     * @param page worker page (StartExam)
     * @param qIndex zero-based question card index ([id^="qcard_"] order)
     * @param answer essay text to fill
     */
    export async function fillEssayAnswer(page: Page, qIndex: number, answer: string): Promise<void> {
      const qCards = page.locator('[id^="qcard_"]');
      const prevText = await page.evaluate(() => {

        },
        prevText,

     *  - .essay-score-input[data-question-id][data-session-id][max]  — score input
     *  - .btn-save-essay-score[data-question-id][data-session-id]    — per-question save
     *  - .btn-finalize-grading[data-session-id]                       — finalize session (confirm dialog)
     *
     * @param pageHc HC user page
     * @param opts {title, category, scheduleDate, sessionId, score} — score 0..maxScore per question
     */
    export async function gradeSingleEssaySession(pageHc: Page, opts: {
      title: string;
      category: string;
      scheduleDate: string;
      sessionId: number;
      score: number;
    }): Promise<void> {
      const baseUrl = pageHc.url() || 'http://localhost:5277';
      const url = new URL('/Admin/AssessmentMonitoringDetail', baseUrl);
      url.searchParams.set('title', opts.title);

      await pageHc.waitForLoadState('networkidle');
