    ```csharp
    // === Parallel PNG pre-compute (REQ EXP-08 — <30s untuk 50 peserta) ===
    // CPU-bound rendering paralel; cap concurrency = ProcessorCount supaya tidak starvation thread pool.
    // CONTEXT D-08: Pakai ConcurrentDictionary untuk hilangkan lock boilerplate (Claude's Discretion).
    var pngCache = new System.Collections.Concurrent.ConcurrentDictionary<int, byte[]>();
    var pngTasks = eligibleSessions
        .Where(s => !s.IsManualEntry)
        .Select(s => new
        {
            SessionId = s.Id,
            EtData = allEtScores
                .Where(et => et.AssessmentSessionId == s.Id)
                .OrderBy(e => e.ElemenTeknis)
                .Select(e => (e.ElemenTeknis, e.QuestionCount > 0 ? (double)e.CorrectCount / e.QuestionCount * 100 : 0d))
                .ToList()
        })
        .Where(x => x.EtData.Count >= 3)
        .ToList();

    await Parallel.ForEachAsync(
        pngTasks,
        new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount },
        (item, ct) =>
        {
            var png = HcPortal.Helpers.SpiderChartRenderer.RenderRadarPng(item.EtData);
            pngCache[item.SessionId] = png;
            return ValueTask.CompletedTask;
        });
    ```

    ```typescript
    import { test, expect } from '@playwright/test';

    // Helper login (sederhana, sesuai pattern kalau existing helper sudah ada).
    async function login(page, username: string, password: string) {
      await page.goto('/Account/Login');
      await page.fill('input[name="Email"]', username);
      await page.fill('input[name="Password"]', password);
      await page.click('button[type="submit"]');
      await page.waitForURL(/\/Home|\/Admin|\/AssessmentAdmin/);
    }

    const ADMIN_USER = process.env.DEV_ADMIN_EMAIL ?? 'admin@pertamina.com';
    const ADMIN_PASS = process.env.DEV_ADMIN_PASSWORD ?? '<set di .env>';
    const HC_USER = process.env.DEV_HC_EMAIL ?? '<hc email>';
    const HC_PASS = process.env.DEV_HC_PASSWORD ?? '<set di .env>';
    const WORKER_USER = process.env.DEV_WORKER_EMAIL ?? '<worker email>';
    const WORKER_PASS = process.env.DEV_WORKER_PASSWORD ?? '<set di .env>';

    // Set di .env atau hardcode lokal saat lokal-only run.
    const EXPORT_URL = '/AssessmentAdmin/ExportAssessmentResults?title=<TitleSeed>&category=<CategorySeed>&scheduleDate=2026-05-15';

    test.describe('Phase 320 Export Per-Peserta — Playwright hybrid', () => {

      test('Admin: trigger Export → assert .xlsx download + content-type', async ({ page }) => {
        await login(page, ADMIN_USER, ADMIN_PASS);
        const downloadPromise = page.waitForEvent('download');
        await page.goto(EXPORT_URL);
        const download = await downloadPromise;
        const path = await download.path();
        expect(path).toBeTruthy();
        expect(download.suggestedFilename()).toMatch(/_Summary\.xlsx$/);
        const fs = await import('node:fs');
        const stat = fs.statSync(path!);
        expect(stat.size).toBeGreaterThan(1024); // > 1KB sanity
      });

      test('HC: trigger Export → assert .xlsx download (REQ EXP-07 HC parity)', async ({ page }) => {
        await login(page, HC_USER, HC_PASS);
        const downloadPromise = page.waitForEvent('download');
        await page.goto(EXPORT_URL);
        const download = await downloadPromise;
        expect(await download.path()).toBeTruthy();
        expect(download.suggestedFilename()).toMatch(/_Summary\.xlsx$/);
      });

      test('Worker: trigger Export → assert 403 atau redirect ke login (REQ EXP-07 negative)', async ({ page }) => {
        await login(page, WORKER_USER, WORKER_PASS);
        const response = await page.goto(EXPORT_URL);
        // Expect either 403 status atau redirect ke /Account/AccessDenied atau /Account/Login
        expect(
          response?.status() === 403 ||
          page.url().includes('/Account/AccessDenied') ||
          page.url().includes('/Account/Login')
        ).toBeTruthy();
      });

      test('Benchmark 50 peserta: response time <30s (REQ EXP-08 SLA)', async ({ page }) => {
        await login(page, ADMIN_USER, ADMIN_PASS);
        const BENCHMARK_URL = process.env.DEV_BENCHMARK_EXPORT_URL ?? EXPORT_URL;
        const start = Date.now();
        const downloadPromise = page.waitForEvent('download', { timeout: 60_000 });
        await page.goto(BENCHMARK_URL);
        const download = await downloadPromise;
        const elapsed = Date.now() - start;
        expect(await download.path()).toBeTruthy();
        expect(elapsed).toBeLessThan(30_000); // REQ EXP-08 gate
        console.log(`Benchmark elapsed: ${elapsed}ms (gate 30000ms)`);
      });
    });
    ```
