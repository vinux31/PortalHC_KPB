// GET /Admin/BulkBackfill — form upload UI
[Authorize(Roles = "Admin")]
public IActionResult BulkBackfill() { return View(); }

// POST /Admin/BulkBackfillAssessment — REST-04 D-04 Hybrid A3
[HttpPost]
[Authorize(Roles = "Admin")]
[ValidateAntiForgeryToken]
[RequestFormLimits(MultipartBodyLengthLimit = 10 * 1024 * 1024)]  // 10 MB cap
public async Task<IActionResult> BulkBackfillAssessment(
    IFormFile excel,
    string auditTag = "ManualImport-Backfill",
    DateTime? completedAtOverride = null,
    int? linkedGroupId = null,
    string? title = null,
    string? category = null,
    string? location = null)
{
    if (excel == null || excel.Length == 0)
    {
        TempData["Error"] = "File Excel wajib diupload.";
        return RedirectToAction(nameof(BulkBackfill));
    }

    var rows = new List<BackfillRow>();   // model: NIP, Score, IsPassed?
    using (var stream = excel.OpenReadStream())
    using (var workbook = new ClosedXML.Excel.XLWorkbook(stream))
    {
        var ws = workbook.Worksheets.First();
        // Parse: assume Sheet 1 header row 1, data row 2+: NIP | Nama | Score | ...
        // Iterate rows starting row 2 until empty
        for (int rowIdx = 2; rowIdx <= ws.LastRowUsed().RowNumber(); rowIdx++)
        {
            var nip = ws.Cell(rowIdx, 1).GetString().Trim();
            if (string.IsNullOrEmpty(nip)) break;
            var score = (int)ws.Cell(rowIdx, 3).GetDouble();
            rows.Add(new BackfillRow { NIP = nip, Score = score });
        }
    }

    if (rows.Count == 0)
    {
        TempData["Error"] = "Tidak ada data terbaca dari Excel.";
        return RedirectToAction(nameof(BulkBackfill));
    }

    // Lookup UserId per NIP
    var nipsLookup = rows.Select(r => r.NIP).ToList();
    var users = await _context.Users.Where(u => nipsLookup.Contains(u.NIP)).ToDictionaryAsync(u => u.NIP!);

    int success = 0;
    var errors = new List<string>();
    var completedAt = completedAtOverride ?? new DateTime(2026, 3, 30, 9, 0, 0, DateTimeKind.Utc);

    using var transaction = await _context.Database.BeginTransactionAsync();
    try
    {
        foreach (var row in rows)
        {
            if (!users.TryGetValue(row.NIP, out var user))
            {
                errors.Add($"NIP {row.NIP} tidak ditemukan di AspNetUsers");
                continue;
            }
            var session = new AssessmentSession
            {
                UserId = user.Id,
                Title = title ?? "PreTest OJT GAST GTO SRU di Unit RU IV Cilacap",
                Category = category ?? "OJT GAST",
                Score = row.Score,
                IsPassed = row.Score >= 70,
                Status = "Closed",
                CompletedAt = completedAt,
                Schedule = completedAt.AddDays(-1),
                IsManualEntry = true,
                LinkedGroupId = linkedGroupId,
                // Other required fields...
            };
            _context.AssessmentSessions.Add(session);

            await _context.AuditLogs.AddAsync(new AuditLog
            {
                UserId = User.FindFirstValue(ClaimTypes.NameIdentifier),
                Action = $"{auditTag} — Insert PreTest backfill",
                EntityType = "AssessmentSession",
                EntityId = 0,  // Updated post-SaveChanges
                Details = $"NIP={row.NIP}, Score={row.Score}, CompletedAt={completedAt:yyyy-MM-dd}",
                Timestamp = DateTime.UtcNow
            });
            success++;
        }

        if (errors.Count > 0)
        {
            await transaction.RollbackAsync();
            TempData["Error"] = $"Rollback — {errors.Count} error: " + string.Join("; ", errors);
            return RedirectToAction(nameof(BulkBackfill));
        }

        await _context.SaveChangesAsync();
        await transaction.CommitAsync();
        TempData["Success"] = $"Berhasil insert {success} backfill assessment dengan tag '{auditTag}'.";
        return RedirectToAction(nameof(BulkBackfill));
    }
    catch (Exception ex)
    {
        await transaction.RollbackAsync();
        TempData["Error"] = $"Bulk backfill gagal: {ex.Message}";
        return RedirectToAction(nameof(BulkBackfill));
    }
}

private class BackfillRow { public string NIP { get; set; } = ""; public int Score { get; set; } }

[HttpGet]
[Authorize(Roles = "Admin, HC")]
public async Task<IActionResult> BulkExportPdf(string title, string category, DateTime scheduleDate)
{
    // Reuse query pattern ExportAssessmentResults L4080-4253 untuk identify eligibleSessions
    var sessions = await _context.AssessmentSessions
        .Include(s => s.User)
        .Where(s => s.Title == title && s.Category == category &&
                    s.Schedule.HasValue && s.Schedule.Value.Date == scheduleDate.Date)
        .ToListAsync();

    var eligibleSessions = sessions
        .Where(s => s.Status != "Cancelled" && (s.CompletedAt != null || s.Score != null))
        .OrderBy(s => s.User?.FullName ?? "")
        .ToList();

    if (eligibleSessions.Count == 0)
        return NotFound("Tidak ada peserta untuk export.");

    // T-338-02 DoS guard: max 50 peserta
    if (eligibleSessions.Count > 50)
        return BadRequest($"Max 50 peserta per batch. Found {eligibleSessions.Count}. Filter lebih spesifik.");

    // Pre-load responses, etScores, questions (same pattern Plan 03)
    var sessionIds = eligibleSessions.Select(s => s.Id).ToList();
    var allResponses = await _context.PackageUserResponses
        .Where(r => sessionIds.Contains(r.AssessmentSessionId)).ToListAsync();
    var allEtScores = await _context.SessionElemenTeknisScores
        .Where(et => sessionIds.Contains(et.AssessmentSessionId)).ToListAsync();
    var sessionPackageMap = await _context.UserPackageAssignments
        .Where(a => sessionIds.Contains(a.AssessmentSessionId))
        .Select(a => new { a.AssessmentSessionId, a.AssessmentPackageId })
        .ToListAsync();
    var packageIds = sessionPackageMap.Select(x => x.AssessmentPackageId).Distinct().ToList();
    var allQuestions = await _context.PackageQuestions
        .Include(q => q.Options)
        .Where(q => packageIds.Contains(q.AssessmentPackageId))
        .ToListAsync();

    // Stream ZIP
    var titleSlug = SlugifyForFilename(title);
    var fileName = $"{titleSlug}_{category}_{scheduleDate:yyyyMMdd}_Bundle.zip";

    var memoryStream = new MemoryStream();
    using (var zipArchive = new System.IO.Compression.ZipArchive(memoryStream, System.IO.Compression.ZipArchiveMode.Create, leaveOpen: true))
    {
        foreach (var session in eligibleSessions)
        {
            var pdfBytes = GeneratePerPesertaPdf(session, allResponses, allEtScores, allQuestions);
            var nameSlug = SlugifyForFilename(session.User?.FullName ?? session.UserId);
            var nip = session.User?.NIP ?? "noNIP";
            var entryName = $"{nip}_{nameSlug}_{titleSlug}.pdf";
            var entry = zipArchive.CreateEntry(entryName, System.IO.Compression.CompressionLevel.Optimal);
            using var entryStream = entry.Open();
            entryStream.Write(pdfBytes, 0, pdfBytes.Length);
        }
    }

    memoryStream.Position = 0;
    return File(memoryStream, "application/zip", fileName);
}

private byte[] GeneratePerPesertaPdf(
    AssessmentSession session,
    List<PackageUserResponse> allResponses,
    List<SessionElemenTeknisScore> allEtScores,
    List<PackageQuestion> allQuestions)
{
    var sessionResponses = allResponses.Where(r => r.AssessmentSessionId == session.Id).ToList();
    var sessionEt = allEtScores.Where(et => et.AssessmentSessionId == session.Id).ToList();

    // Spider chart PNG via Phase 320 Helpers/SpiderChartRenderer (OQ-338-1 resolved)
    byte[]? spiderPng = null;
    if (sessionEt.Any())
    {
        var etData = sessionEt.Select(e => new HcPortal.Models.EtDataPoint
        {
            Label = e.ElemenTeknis,
            Value = (double)e.Score
        }).ToList();
        spiderPng = HcPortal.Helpers.SpiderChartRenderer.RenderRadarPng(etData);
    }

    return QuestPDF.Fluent.Document.Create(container =>
    {
        // Page 1: Cover + Spider Chart
        container.Page(page =>
        {
            page.Size(QuestPDF.Helpers.PageSizes.A4);
            page.Margin(2, QuestPDF.Infrastructure.Unit.Centimetre);
            page.Header().Text(session.Title ?? "Hasil Assessment").FontSize(16).Bold();
            page.Content().Column(col =>
            {
                col.Item().Text($"Nama: {session.User?.FullName ?? "—"}");
                col.Item().Text($"NIP: {session.User?.NIP ?? "—"}");
                col.Item().Text($"Tanggal: {session.CompletedAt:dd MMM yyyy}");
                col.Item().Text($"Skor: {session.Score ?? 0}");
                col.Item().Text($"Status: {(session.IsPassed == true ? "PASSED" : session.IsPassed == false ? "FAILED" : "COMPLETED")}");
                col.Item().PaddingTop(20);
                if (spiderPng != null)
                {
                    col.Item().Image(spiderPng).FitWidth();
                }
            });
        });

        // Page 2+: Jawaban per soal
        if (sessionResponses.Any())
        {
            container.Page(page =>
            {
                page.Size(QuestPDF.Helpers.PageSizes.A4);
                page.Margin(2, QuestPDF.Infrastructure.Unit.Centimetre);
                page.Header().Text("Detail Jawaban Per Soal").FontSize(14).Bold();
                page.Content().Column(col =>
                {
                    int qIdx = 1;
                    foreach (var resp in sessionResponses.OrderBy(r => r.PackageQuestionId))
                    {
                        var q = allQuestions.FirstOrDefault(x => x.Id == resp.PackageQuestionId);
                        if (q == null) continue;
                        col.Item().PaddingVertical(5).Column(qCol =>
                        {
                            qCol.Item().Text($"Soal {qIdx}: {q.QuestionText}").Bold();
                            var ansText = q.Options?.FirstOrDefault(o => o.Id == resp.SelectedOptionId)?.OptionText
                                       ?? resp.AnswerText ?? "—";
                            var correctColor = resp.IsCorrect == true ? QuestPDF.Helpers.Colors.Green.Darken1
                                            : resp.IsCorrect == false ? QuestPDF.Helpers.Colors.Red.Darken1
                                            : QuestPDF.Helpers.Colors.Grey.Darken1;
                            qCol.Item().Text($"Jawaban: {ansText}").FontColor(correctColor);
                        });
                        qIdx++;
                    }
                });
            });
        }
    }).GeneratePdf();
}

private static string SlugifyForFilename(string input)
{
    if (string.IsNullOrEmpty(input)) return "untitled";
    var slug = System.Text.RegularExpressions.Regex.Replace(input, @"[^a-zA-Z0-9]+", "_");
    return slug.Trim('_').Substring(0, Math.Min(50, slug.Length));
}

    [HttpGet]
    [Authorize(Roles = "Admin, HC")]
    public async Task<IActionResult> BulkExportPdf(string title, string category, DateTime scheduleDate)
    {
        if (string.IsNullOrEmpty(title) || string.IsNullOrEmpty(category))
            return BadRequest("title + category mandatory");

        var sessions = await _context.AssessmentSessions
            .Include(s => s.User)
            .Where(s => s.Title == title && s.Category == category &&
                        s.Schedule.HasValue && s.Schedule.Value.Date == scheduleDate.Date)
            .ToListAsync();

        var eligibleSessions = sessions
            .Where(s => s.Status != "Cancelled" && (s.CompletedAt != null || s.Score != null))
            .OrderBy(s => s.User?.FullName ?? "")
            .ToList();

        if (eligibleSessions.Count == 0) return NotFound("Tidak ada peserta untuk export.");
        if (eligibleSessions.Count > 50)
            return BadRequest($"Max 50 peserta per batch. Found {eligibleSessions.Count}.");

        var sessionIds = eligibleSessions.Select(s => s.Id).ToList();
        var allResponses = await _context.PackageUserResponses
            .Where(r => sessionIds.Contains(r.AssessmentSessionId)).ToListAsync();
        var allEtScores = await _context.SessionElemenTeknisScores
            .Where(et => sessionIds.Contains(et.AssessmentSessionId)).ToListAsync();
        var sessionPackageMap = await _context.UserPackageAssignments
            .Where(a => sessionIds.Contains(a.AssessmentSessionId))
            .Select(a => new { a.AssessmentSessionId, a.AssessmentPackageId })
            .ToListAsync();
        var packageIds = sessionPackageMap.Select(x => x.AssessmentPackageId).Distinct().ToList();
        var allQuestions = await _context.PackageQuestions
            .Include(q => q.Options)
            .Where(q => packageIds.Contains(q.AssessmentPackageId))
            .ToListAsync();

        var titleSlug = SlugifyForFilename(title);
        var fileName = $"{titleSlug}_{category}_{scheduleDate:yyyyMMdd}_Bundle.zip";

        var memoryStream = new MemoryStream();
        using (var zipArchive = new System.IO.Compression.ZipArchive(
            memoryStream, System.IO.Compression.ZipArchiveMode.Create, leaveOpen: true))
        {
            foreach (var session in eligibleSessions)
            {
                var pdfBytes = GeneratePerPesertaPdf(session, allResponses, allEtScores, allQuestions);
                var nameSlug = SlugifyForFilename(session.User?.FullName ?? session.UserId);
                var nip = session.User?.NIP ?? "noNIP";
                var entryName = $"{nip}_{nameSlug}_{titleSlug}.pdf";
                var entry = zipArchive.CreateEntry(entryName, System.IO.Compression.CompressionLevel.Optimal);
                using var entryStream = entry.Open();
                entryStream.Write(pdfBytes, 0, pdfBytes.Length);
            }
        }

        memoryStream.Position = 0;
        return File(memoryStream, "application/zip", fileName);
    }

    private byte[] GeneratePerPesertaPdf(
        AssessmentSession session,
        List<PackageUserResponse> allResponses,
        List<SessionElemenTeknisScore> allEtScores,
        List<PackageQuestion> allQuestions)
    {
        var sessionResponses = allResponses.Where(r => r.AssessmentSessionId == session.Id).ToList();
        var sessionEt = allEtScores.Where(et => et.AssessmentSessionId == session.Id).ToList();

        byte[]? spiderPng = null;
        if (sessionEt.Any())
        {
            var etData = sessionEt.Select(e => new HcPortal.Models.EtDataPoint
            {
                Label = e.ElemenTeknis,
                Value = (double)e.Score
            }).ToList();
            spiderPng = HcPortal.Helpers.SpiderChartRenderer.RenderRadarPng(etData);
        }

        return QuestPDF.Fluent.Document.Create(container =>
        {
            container.Page(page =>
            {
                page.Size(QuestPDF.Helpers.PageSizes.A4);
                page.Margin(2, QuestPDF.Infrastructure.Unit.Centimetre);
                page.Header().Text(session.Title ?? "Hasil Assessment").FontSize(16).Bold();
                page.Content().Column(col =>
                {
                    col.Item().Text($"Nama: {session.User?.FullName ?? "—"}");
                    col.Item().Text($"NIP: {session.User?.NIP ?? "—"}");
                    col.Item().Text($"Tanggal: {session.CompletedAt:dd MMM yyyy}");
                    col.Item().Text($"Skor: {session.Score ?? 0}");
                    var statusText = session.IsPassed == true ? "PASSED"
                                   : session.IsPassed == false ? "FAILED" : "COMPLETED";
                    col.Item().Text($"Status: {statusText}").Bold();
                    col.Item().PaddingTop(20);
                    if (spiderPng != null)
                        col.Item().Image(spiderPng).FitWidth();
                });
            });

            if (sessionResponses.Any())
            {
                container.Page(page =>
                {
                    page.Size(QuestPDF.Helpers.PageSizes.A4);
                    page.Margin(2, QuestPDF.Infrastructure.Unit.Centimetre);
                    page.Header().Text("Detail Jawaban Per Soal").FontSize(14).Bold();
                    page.Content().Column(col =>
                    {
                        int qIdx = 1;
                        foreach (var resp in sessionResponses.OrderBy(r => r.PackageQuestionId))
                        {
                            var q = allQuestions.FirstOrDefault(x => x.Id == resp.PackageQuestionId);
                            if (q == null) continue;
                            col.Item().PaddingVertical(5).Column(qCol =>
                            {
                                qCol.Item().Text($"Soal {qIdx}: {q.QuestionText}").Bold();
                                var ansText = q.Options?.FirstOrDefault(o => o.Id == resp.SelectedOptionId)?.OptionText
                                           ?? resp.AnswerText ?? "—";
                                var correctColor = resp.IsCorrect == true ? QuestPDF.Helpers.Colors.Green.Darken1
                                                : resp.IsCorrect == false ? QuestPDF.Helpers.Colors.Red.Darken1
                                                : QuestPDF.Helpers.Colors.Grey.Darken1;
                                qCol.Item().Text($"Jawaban: {ansText}").FontColor(correctColor);
                            });
                            qIdx++;
                        }
                    });
                });
            }
        }).GeneratePdf();
    }

    private static string SlugifyForFilename(string input)
    {
        if (string.IsNullOrEmpty(input)) return "untitled";
        var slug = System.Text.RegularExpressions.Regex.Replace(input, @"[^a-zA-Z0-9]+", "_");
        slug = slug.Trim('_');
        return slug.Length > 50 ? slug.Substring(0, 50) : slug;
    }
