// Helper method baru (private):
private async Task<int?> TryAutoDetectCounterpartGroup(string title, string? category)
{
    // Parse pattern: {Stage} Test {Track} {Lokasi}
    // e.g., "PreTest OJT GAST GTO SRU di Unit RU IV Cilacap"
    //        "PostTest OJT GAST GTO SRU di Unit RU IV Cilacap"
    var match = System.Text.RegularExpressions.Regex.Match(
        title.Trim(),
        @"^(?<stage>Pre|Post)Test\s+(?<rest>.+)$",
        System.Text.RegularExpressions.RegexOptions.IgnoreCase);

    if (!match.Success) return null;

    var stage = match.Groups["stage"].Value;
    var rest = match.Groups["rest"].Value;
    var oppositeStage = stage.Equals("Pre", StringComparison.OrdinalIgnoreCase) ? "Post" : "Pre";

    // Cari session existing di DB dengan title opposite stage + rest same
    var counterpartTitle = $"{oppositeStage}Test {rest}";
    var counterpart = await _context.AssessmentSessions
        .Where(s => s.Title == counterpartTitle)
        .Where(s => category == null || s.Category == category)
        .Where(s => s.LinkedGroupId != null)  // Counterpart sudah punya GroupId
        .Select(s => s.LinkedGroupId)
        .FirstOrDefaultAsync();

    return counterpart;
}
```

NOTE: Bila counterpart belum punya LinkedGroupId, create new GroupId via max + 1 logic atau biarkan null (admin manual set).

<#
.SYNOPSIS
    SQL Server Database backup pre-migration. Phase 338 REST-05.

.DESCRIPTION
    Standalone PowerShell script untuk IT jalankan SEBELUM migration.
    Tidak hardcode credential — semua via param atau Windows Auth.

.PARAMETER Server
    SQL Server instance (e.g., "10.55.3.3" atau "(localdb)\MSSQLLocalDB").

.PARAMETER Database
    Database name (e.g., "HcPortalDB_Dev").

.PARAMETER OutputPath
    Full path output .bak file. Folder harus exist.

.EXAMPLE
    .\backup-dev-pre-migration.ps1 -Server "10.55.3.3" -Database "HcPortalDB_Dev" -OutputPath "C:\Backup\HcPortalDB_Dev_pre_20260530_143000.bak"
#>

param(
    [Parameter(Mandatory=$true)] [string]$Server,
    [Parameter(Mandatory=$true)] [string]$Database,
    [Parameter(Mandatory=$true)] [string]$OutputPath
)

$ErrorActionPreference = 'Stop'

# Validate output folder exists
$folder = Split-Path -Parent $OutputPath
if (-not (Test-Path $folder)) {
    Write-Error "Output folder tidak ada: $folder. Create dulu lalu retry."
    exit 1
}

# Validate sqlcmd available
$sqlcmd = Get-Command sqlcmd -ErrorAction SilentlyContinue
if (-not $sqlcmd) {
    Write-Error "sqlcmd tidak ditemukan di PATH. Install SQL Server Command Line Utilities."
    exit 1
}

$query = @"
BACKUP DATABASE [$Database]
TO DISK = N'$OutputPath'
WITH FORMAT, INIT, COMPRESSION,
NAME = N'$Database Pre-Migration Full Backup',
STATS = 10;
"@

Write-Host "Backup starting: $Server / $Database -> $OutputPath" -ForegroundColor Cyan

& sqlcmd -S $Server -E -Q $query

if ($LASTEXITCODE -ne 0) {
    Write-Error "Backup gagal (exit code $LASTEXITCODE). Check sqlcmd output above."
    exit $LASTEXITCODE
}

# Verify file created
if (Test-Path $OutputPath) {
    $size = (Get-Item $OutputPath).Length / 1MB
    Write-Host ("Backup SUKSES: {0:N1} MB at {1}" -f $size, $OutputPath) -ForegroundColor Green
} else {
    Write-Error "Backup file tidak terbuat. Check SQL Server log."
    exit 1
}
```
