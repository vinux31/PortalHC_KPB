    // owner ∥ roleLevel<=3 (Admin/HC/L3) ∥ (L4 section-scoped, Section non-null).
    public static bool IsResultsAuthorized(string? ownerUserId, string currentUserId, int roleLevel, string? currentUserSection, string? ownerSection)
    {
        if (ownerUserId == currentUserId) return true;          // owner
        if (roleLevel <= 3) return true;                         // Admin(1)/HC(2)/Direktur-VP-Manager(3)
        if (roleLevel == 4
            && !string.IsNullOrEmpty(currentUserSection)         // guard: L4 Section null/empty tidak lolos

        return false;                                            // L5/L6 non-owner, L4 beda/null section
    }
