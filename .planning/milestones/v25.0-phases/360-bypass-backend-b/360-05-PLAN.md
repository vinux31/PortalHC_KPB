    ```
    var pending = FindAsync(pendingId);
    if (pending==null || (pending.Status!="Menunggu" && pending.Status!="Siap"))
        return BypassResult(false, "Pending tidak bisa dibatalkan (status tidak valid).");
    await using tx = BeginTransactionAsync;
    try {
      // Simpan status SEBELUM update:
      bool workerLulus = pending.Status == "Siap";  // status SEBELUM update = Siap → worker sudah lulus
      var linkedId = pending.LinkedAssessmentSessionId;

      // Atomik set Dibatalkan (anti klik-dobel):
      var rows = await _context.PendingProtonBypasses
          .Where(p => p.Id==pendingId && (p.Status=="Menunggu" || p.Status=="Siap"))
          .ExecuteUpdateAsync(p => p.SetProperty(x=>x.Status,"Dibatalkan").SetProperty(x=>x.ResolvedAt, DateTime.UtcNow));
      if (rows==0) { rollback; return BypassResult(false,"Pending sudah diproses."); }

      // Auto-cancel LinkedAssessmentSession:
      if (workerLulus) {
         // PERTAHANKAN hasil exam (penanda Origin="Exam" sah) — JANGAN sentuh session. Hanya batalkan rencana.
      } else {
         // worker BELUM lulus (pending "Menunggu"). W-03 GUARD: cancel sesi HANYA bila session belum Completed.
         //   Worker yang sudah mengerjakan-tapi-GAGAL → session Status="Completed" (IsPassed=false) → JANGAN overwrite
         //   "Dibatalkan" (itu jejak worker pernah kerjakan + gagal, spec §11 MISS-2). Hanya sesi yang benar-benar
         //   belum dikerjakan (Status != "Completed") yang di-set "Dibatalkan".
         await _context.AssessmentSessions
            .Where(s=>s.Id==linkedId && s.Status != "Completed")
            .ExecuteUpdateAsync(s=>s.SetProperty(x=>x.Status,"Dibatalkan"));
      }
      // Force-approve deliverable source TETAP (jejak history) — JANGAN revert.
      // audit log.
      SaveChanges + Commit; return BypassResult(true,"Rencana bypass dibatalkan.");
    } catch (...) rollback + ramah tanpa ex.Message;
    ```
