public static string DeriveUserStatus(string? status, DateTime? completedAt, DateTime? startedAt)
{
    if (status == AssessmentConstants.AssessmentStatus.PendingGrading) return "Menunggu Penilaian";
    if (completedAt != null) return "Completed";
    if (status == "Cancelled") return "Dibatalkan";
    if (status == "Abandoned") return "Abandoned";

}
