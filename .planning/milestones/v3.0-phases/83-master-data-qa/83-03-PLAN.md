```csharp
// POST: /ProtonData/SilabusDeactivate — soft delete a Kompetensi (sets IsActive=false)
[HttpPost]
[Authorize(Roles = "Admin, HC")]
[ValidateAntiForgeryToken]
public async Task<IActionResult> SilabusDeactivate([FromBody] int kompetensiId)
{
    var user = await _userManager.GetUserAsync(User);
    if (user == null) return Challenge();

    if (kompetensiId <= 0)
        return Json(new { success = false, message = "ID tidak valid." });

    var komp = await _context.ProtonKompetensiList.FindAsync(kompetensiId);
    if (komp == null)
        return Json(new { success = false, message = "Data tidak ditemukan." });

    if (!komp.IsActive)
        return Json(new { success = false, message = "Kompetensi ini sudah tidak aktif." });

    komp.IsActive = false;
    await _context.SaveChangesAsync();

    await _auditLog.LogAsync(user.Id, user.FullName, "Deactivate",
        $"Deactivated silabus kompetensi '{komp.NamaKompetensi}' (ID {komp.Id})",
        targetId: komp.Id, targetType: "ProtonKompetensi");

    return Json(new { success = true, message = $"Silabus '{komp.NamaKompetensi}' berhasil dinonaktifkan." });
}
```

// POST: /ProtonData/SilabusReactivate — restore a deactivated Kompetensi (sets IsActive=true)
[HttpPost]
[Authorize(Roles = "Admin, HC")]
[ValidateAntiForgeryToken]
public async Task<IActionResult> SilabusReactivate([FromBody] int kompetensiId)
{
    var user = await _userManager.GetUserAsync(User);
    if (user == null) return Challenge();

    if (kompetensiId <= 0)
        return Json(new { success = false, message = "ID tidak valid." });

    var komp = await _context.ProtonKompetensiList.FindAsync(kompetensiId);
    if (komp == null)
        return Json(new { success = false, message = "Data tidak ditemukan." });

    if (komp.IsActive)
        return Json(new { success = false, message = "Kompetensi ini sudah aktif." });

    komp.IsActive = true;
    await _context.SaveChangesAsync();

    await _auditLog.LogAsync(user.Id, user.FullName, "Reactivate",
        $"Reactivated silabus kompetensi '{komp.NamaKompetensi}' (ID {komp.Id})",
        targetId: komp.Id, targetType: "ProtonKompetensi");

    return Json(new { success = true, message = $"Silabus '{komp.NamaKompetensi}' berhasil diaktifkan kembali." });
}
```

function deactivateKomp(id, name) {
    if (!confirm('Nonaktifkan kompetensi "' + name + '"? Kompetensi ini akan disembunyikan dari Plan IDP dan Coaching Proton.')) return;

    fetch('/ProtonData/SilabusDeactivate', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'RequestVerificationToken': document.querySelector('input[name="__RequestVerificationToken"]').value
        },
        body: JSON.stringify(id)
    })
    .then(r => r.json())
    .then(data => {
        if (data.success) {
            showToast(data.message, 'success');
            setTimeout(() => location.reload(), 1000);
        } else {
            showToast(data.message || 'Gagal menonaktifkan.', 'danger');
        }
    });
}

function reactivateKomp(id) {
    fetch('/ProtonData/SilabusReactivate', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'RequestVerificationToken': document.querySelector('input[name="__RequestVerificationToken"]').value

        body: JSON.stringify(id)
    })
    .then(r => r.json())
    .then(data => {
        if (data.success) {
            showToast(data.message, 'success');
            setTimeout(() => location.reload(), 1000);
        } else {
            showToast(data.message || 'Gagal mengaktifkan.', 'danger');
        }
    });
}
```
