# Plan 89-04: Full Rewrite — CMPController.Kkj() + Views/CMP/Kkj.cshtml

## Goal

Rewrite `CMPController.Kkj()` and `Views/CMP/Kkj.cshtml` to implement:
1. Always show the KKJ matrix view (no redirect to the separate KkjSectionSelect page)
2. Bagian dropdown in the view header — options filtered by role
3. Role-based access: L1/L2/L3/L4 (Admin, HC, Management, SectionHead, SrSupervisor) see ALL Bagians;
   L5/L6 (Coach, Supervisor, Coachee) see only their own Bagian
4. Read-only table with dynamic columns from KkjColumn and dynamic values from KkjTargetValue
5. Clean, modern Bootstrap 5 design matching the Admin KkjMatrix style

The old KkjSectionSelect.cshtml has hardcoded bagian names and is no longer needed as the primary
entry point — the Kkj view itself now handles bagian selection via a dropdown.

## must_haves

- `CMPController.Kkj(string? section)` never redirects to `KkjSectionSelect` — always renders `Kkj` view
- Roles L1-L4 (level <= 4): `ViewBag.AllBagians` contains ALL `KkjBagian` records (from DB) for dropdown
- Roles L5-L6 (level >= 5): `ViewBag.AllBagians` contains only the user's own bagian (matched by `user.Unit`)
- `ViewBag.SelectedSection` = the currently selected bagian name (from `section` param or user.Unit)
- `ViewBag.Columns` = ordered `List<KkjColumn>` for the selected section
- `Model` = `List<KkjMatrixItem>` for the selected section with `.TargetValues` and `.KkjColumn` included
- View renders a Bagian dropdown in the page header (populated from `ViewBag.AllBagians`)
- Changing the dropdown navigates to `?section=X` (form GET or anchor link per bagian)
- View renders a read-only table: sticky header row, first 5 columns sticky, dynamic target value columns
- View has a search/filter input that filters rows by text
- View has crosshair highlighting on hover (row + column)
- Target value cells styled with color scale: 1=red, 2=orange, 3=yellow, 4=light-green, 5=green, -=grey
- `dotnet build` succeeds with 0 errors

```csharp
// --- HALAMAN 1: SUSUNAN KKJ (MATRIX VIEW) ---
public async Task<IActionResult> Kkj(string? section)
{
    // Get current user and role
    var user = await _userManager.GetUserAsync(User);
    var userRoles = user != null ? await _userManager.GetRolesAsync(user) : new List<string>();
    var userRole = userRoles.FirstOrDefault() ?? "";
    int userLevel = user?.RoleLevel ?? 6;

    ViewBag.UserRole = userRole;
    ViewBag.UserLevel = userLevel;

    // Load all bagians from DB for the dropdown
    var allBagians = await _context.KkjBagians
        .OrderBy(b => b.DisplayOrder)
        .ToListAsync();

    // Role-based filter: L1-L4 (level <= 4) see all bagians; L5-L6 see only their own
    List<KkjBagian> availableBagians;
    if (userLevel <= 4)
    {
        // Admin, HC, Management (L3), SrSupervisor (L4) — all bagians
        availableBagians = allBagians;
    }
    else
    {
        // Coach, Supervisor (L5), Coachee (L6) — own bagian only
        var userUnit = user?.Unit ?? "";
        availableBagians = allBagians
            .Where(b => b.Name == userUnit)
            .ToList();
    }

    ViewBag.AllBagians = availableBagians;

    // Determine selected section
    string? selectedSection = section;

    // If no section passed:
    // - L1-L4: default to first available bagian (if any)
    // - L5-L6: use user's own unit
    if (string.IsNullOrEmpty(selectedSection))
    {
        if (userLevel <= 4)
        {
            selectedSection = availableBagians.FirstOrDefault()?.Name;
        }
        else
        {
            selectedSection = user?.Unit;
        }
    }
    else
    {
        // Validate that the requested section is in the user's available bagians
        // (prevent L5/L6 from accessing other bagians by URL manipulation)
        var allowed = availableBagians.Any(b => b.Name == selectedSection);
        if (!allowed)
        {
            // Fall back to own bagian
            selectedSection = userLevel <= 4
                ? availableBagians.FirstOrDefault()?.Name
                : user?.Unit;
        }
    }

    ViewBag.SelectedSection = selectedSection;

    // Load KkjColumns for the selected section
    var bagian = availableBagians.FirstOrDefault(b => b.Name == selectedSection);
    if (bagian != null)
    {
        // Load columns separately (bagian was loaded without Include)
        var columns = await _context.KkjColumns
            .Where(c => c.BagianId == bagian.Id)
            .OrderBy(c => c.DisplayOrder)
            .ToListAsync();
        ViewBag.Columns = columns;
    }
    else
    {
        ViewBag.Columns = new List<KkjColumn>();
    }

    // Load matrix items for selected section with target values
    var matrixData = new List<KkjMatrixItem>();
    if (!string.IsNullOrEmpty(selectedSection))
    {
        matrixData = await _context.KkjMatrices
            .Where(k => k.Bagian == selectedSection)
            .Include(m => m.TargetValues)
            .ThenInclude(v => v.KkjColumn)
            .OrderBy(k => k.No)
            .ToListAsync();
    }

    return View(matrixData);
}
```

```razor
            if (!cell || cell.closest('thead')) return;
            var mRow = cell.closest('tr');
            if (!mRow || mRow.parentElement.tagName !== 'TBODY') return;

            mRow.classList.add('hover-row');

            var cellIndex = cell.cellIndex;
            if (cellIndex >= 5) {
                cell.classList.add('hover-cell');
                table.querySelectorAll('tbody tr').forEach(function (row) {
                    var c = row.children[cellIndex];
                    if (c) c.classList.add('hover-col');
                });
            }

        table.addEventListener('mouseout', function (e) {
            var cell = e.target.closest('td');
            if (!cell) return;
            var mRow = cell.closest('tr');
            if (mRow) mRow.classList.remove('hover-row');
            if (cell.classList.contains('hover-cell')) cell.classList.remove('hover-cell');
            table.querySelectorAll('.hover-col').forEach(function (c) { c.classList.remove('hover-col'); });
        });
    }
});
```
