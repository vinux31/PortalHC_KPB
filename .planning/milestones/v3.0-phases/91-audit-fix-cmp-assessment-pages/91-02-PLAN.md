**StartExam.cshtml — Fix 1: Replace 1x retry with 3-attempt exponential backoff:**
The current `saveAnswerAsync(qId, optId, isRetry)` uses a boolean retry flag allowing 1x retry. Replace with an attempt counter supporting 3 total attempts with exponential backoff delays: 0ms (immediate), 1000ms, 3000ms.

Replace the entire `saveAnswerAsync` function with:
```javascript
function saveAnswerAsync(qId, optId, attempt) {
    if (attempt === undefined) attempt = 0;
    if (inFlightSaves.has(String(qId))) return;
    inFlightSaves.add(String(qId));
    showSaveIndicator(qId, 'saving');

    doSaveFetch(qId, optId)
        .then(function() {
            showSaveIndicator(qId, 'saved');
        })
        .catch(function() {
            inFlightSaves.delete(String(qId));
            if (attempt < 2) {
                var delay = attempt === 0 ? 1000 : 3000;
                setTimeout(function() {
                    saveAnswerAsync(qId, optId, attempt + 1);
                }, delay);
            } else {
                showSaveIndicator(qId, 'error');
                showFailureToast();
            }
        })
        .finally(function() {
            inFlightSaves.delete(String(qId));
        });
}
```
Update the call site in `saveAnswerWithDebounce` from `saveAnswerAsync(qId, optId, false)` to `saveAnswerAsync(qId, optId, 0)`.
