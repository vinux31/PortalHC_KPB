```csharp
[HttpPost]                                  // POST: carries #QuestionsJson + UserIds (too large for query string)
[Authorize(Roles = "Admin, HC")]
[ValidateAntiForgeryToken]
public async Task<IActionResult> DownloadInjectTemplate(InjectAssessmentViewModel vm)
{
    var questionVms = ParseQuestionVms(vm);
    var questions = MapVmQuestionsToSpec(questionVms);   // see note below
    if (questions.Count == 0) { TempData["Error"] = "Belum ada soal. Tambahkan soal di Langkah 3 sebelum membuat template."; ... return RedirectToAction(nameof(InjectAssessment)); }
    var userIds = vm.UserIds ?? new List<string>();
    var workerRows = await _context.Users
        .Where(u => userIds.Contains(u.Id) && u.NIP != null && u.NIP != "")
        .Select(u => new { Nip = u.NIP!, Name = u.FullName ?? "" })
        .ToListAsync();
    if (workerRows.Count == 0) { TempData["Error"] = "Pilih pekerja ber-NIP di Langkah 2 sebelum membuat template."; ... return RedirectToAction(nameof(InjectAssessment)); }
    var workers = workerRows.Select(w => (w.Nip, w.Name)).ToList();
    using var wb = InjectExcelHelper.GenerateTemplate(questions, workers);
    return ExcelExportHelper.ToFileResult(wb, "inject_template.xlsx", this);
}
```

```csharp
[HttpPost]
[Authorize(Roles = "Admin, HC")]
[ValidateAntiForgeryToken]
[RequestFormLimits(MultipartBodyLengthLimit = 10 * 1024 * 1024)]
public async Task<IActionResult> UploadInjectExcel(IFormFile? excel, InjectAssessmentViewModel vm)
{
    var result = new InjectExcelUploadResult();
    // 1. file guards (Security V12): null/empty + extension whitelist
    if (excel == null || excel.Length == 0) { result.Ok=false; result.Errors.Add(new InjectRowError{ Message="File Excel wajib diunggah." }); return Json(result); }
    var ext = System.IO.Path.GetExtension(excel.FileName).ToLowerInvariant();
    if (ext != ".xlsx" && ext != ".xls") { result.Ok=false; result.Errors.Add(new InjectRowError{ Message="Format file harus .xlsx atau .xls." }); return Json(result); }
    // 2. resolve questions + picker audience (D-02): allowedNips = picker NIPs only
    var questions = MapVmQuestionsToSpec(ParseQuestionVms(vm));
    var userIds = vm.UserIds ?? new List<string>();
    var picker = await _context.Users.Where(u => userIds.Contains(u.Id) && u.NIP != null && u.NIP != "")
        .Select(u => new { u.Id, Nip = u.NIP!, Name = u.FullName ?? "" }).ToListAsync();
    var allowedNips = picker.Select(p => p.Nip).ToHashSet();
    var nipToUserId = picker.ToDictionary(p => p.Nip, p => p.Id);
    var nipToName = picker.ToDictionary(p => p.Nip, p => p.Name);
    // 3. parse (try/catch inside helper → never 500)
    using var stream = excel.OpenReadStream();
    var (workers, errors, skippedBlank) = InjectExcelHelper.ParseMatrix(stream, questions, allowedNips, nipToUserId);
    // 4. atomic D-09: ANY error → reject-all, NO write, full list
    if (errors.Count > 0) { result.Ok=false; result.Errors=errors; return Json(result); }
    // 5. preview each worker via the SAME engine as commit (D-08, NO cert#)
    result.Ok = true;
    result.SkippedBlankCount = skippedBlank;
    result.Previews = BuildExcelPreviews(questions, workers, nipToName, vm.PassPercentage, nipToUserId);
    result.AnswersJson = System.Text.Json.JsonSerializer.Serialize(workers);
    return Json(result);
}
```
Add private `BuildExcelPreviews(...)`: loop `workers`; for each, convert its `InjectAnswerVM` list to `InjectAnswerSpec` (reuse the existing `ToAnswerSpec` projection — make it usable here or replicate), call `MapToInMemory(questions, answerSpecs)` then `AssessmentScoreAggregator.Compute(qInMem, respInMem, passPercentage)`. Build an `InjectExcelPreviewRow` with Nip (reverse-lookup from nipToUserId/UserId), Name (nipToName), Percentage, IsPassed, Answered = number of answers actually present, TotalQuestions = questions.Count. Return `List<InjectExcelPreviewRow>`. NO `CertNumberHelper`, NO `SaveChanges`.

These return `Json(...)` so the `~/Views/Admin/` View() override does not interfere (same reason as PreviewInjectScore comment). Wrap any outer exception path into a friendly error result, never let a 500 leak (Security V5).
