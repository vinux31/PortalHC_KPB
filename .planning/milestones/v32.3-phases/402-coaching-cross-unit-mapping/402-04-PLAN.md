function applyCoachScope() {
    var sel = document.getElementById('assignCoachSelect');
    var opt = sel.options[sel.selectedIndex];
    var coachSection = (opt && opt.getAttribute('data-section')) || '';
    var prompt = document.getElementById('coacheePrompt');
    var filterGroup = document.getElementById('coacheeSectionFilterGroup');
    var items = document.querySelectorAll('#coacheeChecklist .coachee-item');
    var secSelect = document.getElementById('assignAssignmentSection');

    // Reset selections when coach changes (IC-1: ganti coach => uncheck + re-filter)
    items.forEach(function (item) { var cb = item.querySelector('.coachee-checkbox'); if (cb) cb.checked = false; });

    if (!coachSection) {
        // No coach: show prompt, hide all coachee-items, neutral filters
        if (prompt) prompt.style.display = '';
        items.forEach(function (item) { item.style.display = 'none'; });
        if (filterGroup) filterGroup.style.display = '';
        if (secSelect) { secSelect.disabled = false; secSelect.value = ''; }
        return;
    }
    if (prompt) prompt.style.display = 'none';
    // Filter checklist to coach.Section (CXU-01 client; server enforces CXU-02)
    items.forEach(function (item) {

        secSelect.value = coachSection;

function submitAssign() {
    var sel = document.getElementById('assignCoachSelect');
    var coachId = sel.value;
    if (!coachId) { alert('Pilih coach terlebih dahulu.'); return; }
    var opt = sel.options[sel.selectedIndex];
    var coachSection = (opt && opt.getAttribute('data-section')) || '';

    var checked = Array.from(document.querySelectorAll('.coachee-checkbox:checked'));
    if (checked.length === 0) { alert('Pilih minimal satu coachee.'); return; }

    // E-2 backstop (Bagian-level): semua coachee tercentang harus 1 Bagian = coach.Section.
    var crossBagian = checked.some(function (cb) {
        return cb.closest('.coachee-item').getAttribute('data-section') !== coachSection;
    });
    if (crossBagian) {
        alert('Semua coachee harus berada dalam ' + '@OrgLabels.GetLabel(0)' + ' yang sama dengan coach. Periksa kembali pilihan Anda.');
        return;
    }

    // Build per-coachee unit map (IC-4): per-row select (multi-unit) or single unit from data-units.
    var assignmentUnits = {};
    for (var i = 0; i < checked.length; i++) {
        var cb = checked[i];
        var coacheeId = cb.value;
        var item = cb.closest('.coachee-item');
        var rowSelect = item.querySelector('.coachee-unit-select');
        if (rowSelect) {
            assignmentUnits[coacheeId] = rowSelect.value;
        } else {
            var units = [];
            try { units = JSON.parse(item.getAttribute('data-units') || '[]'); } catch (e) { units = []; }
            if (units.length === 1) { assignmentUnits[coacheeId] = units[0]; }
        }
        if (!assignmentUnits[coacheeId]) {
            alert('Unit penugasan wajib dipilih untuk setiap coachee multi-unit.');   // E-3
            return;
        }

    var coacheeIds = checked.map(function (cb) { return cb.value; });
    var trackId = document.getElementById('assignProtonTrack').value;
    var startDate = document.getElementById('assignStartDate').value;

    var payload = {
        CoachId: coachId,
        CoacheeIds: coacheeIds,
        ProtonTrackId: trackId ? parseInt(trackId) : null,
        StartDate: startDate || null,
        AssignmentSection: coachSection,           // single = coach.Section (IC-1)
        AssignmentUnits: assignmentUnits           // per-coachee map (IC-4 / D-05)
    };

    fetch(appUrl('/Admin/CoachCoacheeMappingAssign'), {
        method: 'POST',
        headers: {

            'RequestVerificationToken': document.querySelector('[name=__RequestVerificationToken]').value

        if (data.success) { location.reload(); }
        else { alert(data.message); }            // E-1 server message surfaced here
    })
    .catch(err => alert('Terjadi kesalahan: ' + err.message));
