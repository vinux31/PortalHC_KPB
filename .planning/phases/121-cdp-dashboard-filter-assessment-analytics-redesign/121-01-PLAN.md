   ```csharp
   [HttpGet]
   public async Task<IActionResult> FilterCoachingProton(string? section, string? unit, string? category, string? track)
   {
       var user = await _userManager.GetUserAsync(User);
       if (user == null) return Unauthorized();
       var roles = await _userManager.GetRolesAsync(user);
       var userRole = roles.FirstOrDefault() ?? "";

       // Server-side enforcement: override section/unit for restricted roles
       int roleLevel = UserRoles.GetRoleLevel(userRole);
       if (UserRoles.HasSectionAccess(roleLevel)) { section = user.Section; }
       else if (UserRoles.IsCoachingRole(roleLevel)) { section = user.Section; unit = user.Unit; }

       var model = await BuildProtonProgressSubModelAsync(user, userRole, section, unit, category, track);
       return PartialView("Shared/_CoachingProtonPartial", model);
   }
   ```

4. **Add `GetCascadeOptions` JSON endpoint** to CDPController:

   ```csharp
   [HttpGet]
   public IActionResult GetCascadeOptions(string? section, string? unit)
   {
       var units = string.IsNullOrEmpty(section) ? new List<string>() : OrganizationStructure.GetUnitsForSection(section);
       // Categories = distinct TrackTypes from ProtonTracks
       var categories = _context.ProtonTracks.Select(t => t.TrackType).Distinct().OrderBy(t => t).ToList();
       // Tracks = filtered by category if provided
       var tracks = _context.ProtonTracks.OrderBy(t => t.Urutan).Select(t => t.DisplayName).ToList();
       return Json(new { units, categories, tracks });
   }
   ```

     ```javascript
     async function refreshProtonContent() {
         var container = document.getElementById('proton-content');
         container.classList.add('dashboard-loading');
         // Build query string from current dropdown values
         var params = new URLSearchParams();
         if (sectionVal) params.set('section', sectionVal);
         if (unitVal) params.set('unit', unitVal);
         if (categoryVal) params.set('category', categoryVal);
         if (trackVal) params.set('track', trackVal);
         // Abort previous request
         if (protonAbort) protonAbort.abort();
         protonAbort = new AbortController();
         try {
             var resp = await fetch('/CDP/FilterCoachingProton?' + params, { signal: protonAbort.signal });
             var html = await resp.text();
             container.innerHTML = html;
             // Execute script tags in response (for Chart.js)
             container.querySelectorAll('script').forEach(function(oldScript) {
                 var newScript = document.createElement('script');
                 newScript.textContent = oldScript.textContent;
                 oldScript.parentNode.replaceChild(newScript, oldScript);
             });
         } catch(e) { if (e.name !== 'AbortError') console.error(e); }
         container.classList.remove('dashboard-loading');
     }
     ```
