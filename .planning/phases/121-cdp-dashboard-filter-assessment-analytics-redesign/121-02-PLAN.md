   [HttpGet]
   public async Task<IActionResult> FilterAssessmentAnalytics(
       string? section, string? unit, string? category, int page = 1, int pageSize = 20)
   {
       var user = await _userManager.GetUserAsync(User);
       if (user == null) return Unauthorized();
       var roles = await _userManager.GetRolesAsync(user);
       var userRole = roles.FirstOrDefault() ?? "";
       // Only HC/Admin can access
       if (userRole != UserRoles.HC && userRole != UserRoles.Admin) return Forbid();

       var model = await BuildAnalyticsSubModelAsync(section, unit, category, page, pageSize);
       return PartialView("Shared/_AssessmentAnalyticsPartial", model);
   }
   ```

     var analyticsAbort = null;
     async function refreshAnalyticsContent(page) {
         page = page || 1;
         var container = document.getElementById('analytics-content');
         container.classList.add('dashboard-loading');
         var params = new URLSearchParams();
         var section = document.getElementById('analyticsFilterSection').value;
         var unit = document.getElementById('analyticsFilterUnit').value;
         var category = document.getElementById('analyticsFilterCategory').value;
         if (section) params.set('section', section);
         if (unit) params.set('unit', unit);
         if (category) params.set('category', category);
         params.set('page', page);
         if (analyticsAbort) analyticsAbort.abort();
         analyticsAbort = new AbortController();
         try {
             var resp = await fetch('/CDP/FilterAssessmentAnalytics?' + params, { signal: analyticsAbort.signal });
             var html = await resp.text();
             container.innerHTML = html;
             // Re-execute scripts for charts
             container.querySelectorAll('script').forEach(function(old) {
                 var s = document.createElement('script');
                 s.textContent = old.textContent;
                 old.parentNode.replaceChild(s, old);
             });
             // Re-bind pagination clicks
             bindAnalyticsPagination();
         } catch(e) { if (e.name !== 'AbortError') console.error(e); }
         container.classList.remove('dashboard-loading');
     }
     ```
