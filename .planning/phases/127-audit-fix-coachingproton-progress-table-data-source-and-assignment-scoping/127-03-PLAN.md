       foreach (var deliverableId in newDeliverableIds)

           // Check doesn't already exist (belt and suspenders)
           var exists = await _context.ProtonDeliverableProgresses
               .AnyAsync(p => p.ProtonTrackAssignmentId == assignment.Id && p.ProtonDeliverableId == deliverableId);
           if (!exists)

               _context.ProtonDeliverableProgresses.Add(new ProtonDeliverableProgress

                   ProtonDeliverableId = deliverableId,
                   ProtonTrackAssignmentId = assignment.Id,

                   CreatedAt = DateTime.UtcNow

1. In `DeleteKompetensi` (line ~860), the existing cascade delete logic already removes SubKompetensi and Deliverables. Before deleting deliverables, also delete associated progress records:

   ```csharp
   // Get all deliverable IDs that will be deleted (under this kompetensi)
   var deliverableIds = await _context.ProtonDeliverables
       .Where(d => d.ProtonSubKompetensi!.ProtonKompetensiId == kompetensiId)
       .Select(d => d.Id)
       .ToListAsync();

   // Delete progress records referencing these deliverables
   var progressIds = await _context.ProtonDeliverableProgresses
       .Where(p => deliverableIds.Contains(p.ProtonDeliverableId))
       .Select(p => p.Id)
       .ToListAsync();

   if (progressIds.Any())
   {
       // Delete DeliverableStatusHistory
       var histories = await _context.DeliverableStatusHistories
           .Where(h => progressIds.Contains(h.ProtonDeliverableProgressId))
           .ToListAsync();
       _context.DeliverableStatusHistories.RemoveRange(histories);

       // Delete CoachingSessions
       var sessions = await _context.CoachingSessions
           .Where(s => s.ProtonDeliverableProgressId.HasValue && progressIds.Contains(s.ProtonDeliverableProgressId.Value))
           .ToListAsync();
       _context.CoachingSessions.RemoveRange(sessions);

       // Delete progress records
       var progresses = await _context.ProtonDeliverableProgresses
           .Where(p => deliverableIds.Contains(p.ProtonDeliverableId))
           .ToListAsync();
       _context.ProtonDeliverableProgresses.RemoveRange(progresses);
   }
   ```

3. Note: The FK has ON DELETE CASCADE from ProtonTrackAssignment, but ProtonDeliverable FK uses DeleteBehavior.Restrict, so we must manually cascade deletes from deliverable removal.
