```csharp
// Phase 129: Defensive unit filter — remove progress where kompetensi unit doesn't match assignment's resolved unit
var assignmentCoacheeIds = await _context.ProtonTrackAssignments
    .Where(a => activeAssignmentIds.Contains(a.Id))
    .Select(a => new { a.Id, a.CoacheeId })
    .ToListAsync();
var coacheeIdList = assignmentCoacheeIds.Select(x => x.CoacheeId).Distinct().ToList();
var mappingUnits = await _context.CoachCoacheeMappings
    .Where(m => m.IsActive && coacheeIdList.Contains(m.CoacheeId))
    .Select(m => new { m.CoacheeId, m.AssignmentUnit })
    .ToListAsync();
var userUnits = await _context.Users
    .Where(u => coacheeIdList.Contains(u.Id))
    .Select(u => new { u.Id, u.Unit })
    .ToDictionaryAsync(u => u.Id, u => u.Unit);

var assignmentUnitMap = assignmentCoacheeIds.ToDictionary(
    x => x.Id,
    x => (mappingUnits.FirstOrDefault(m => m.CoacheeId == x.CoacheeId)?.AssignmentUnit
          ?? userUnits.GetValueOrDefault(x.CoacheeId))?.Trim() ?? "");

progresses = progresses.Where(p => {
    var kompUnit = p.ProtonDeliverable?.ProtonSubKompetensi?.ProtonKompetensi?.Unit?.Trim() ?? "";
    var resolvedUnit = assignmentUnitMap.GetValueOrDefault(p.ProtonTrackAssignmentId, "");
    return string.IsNullOrEmpty(resolvedUnit) || kompUnit == resolvedUnit;
}).ToList();
```

Place this AFTER `var progresses = await query ... .ToListAsync();` (line ~1308) and BEFORE the coachee name lookup and data building.
