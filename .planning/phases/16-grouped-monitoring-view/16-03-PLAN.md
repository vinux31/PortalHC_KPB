Create `Views/CMP/AssessmentMonitoringDetail.cshtml` with the following structure. Model is `HcPortal.Models.MonitoringGroupViewModel`.

```razor
@model HcPortal.Models.MonitoringGroupViewModel

@{
    ViewData["Title"] = "Assessment Detail — " + Model.Title;

    var catBadgeClass = Model.Category switch
    {
        "OJT"                     => "bg-primary",
        "IHT"                     => "bg-success",
        "Training Licencor"        => "bg-danger",
        "OTS"                     => "bg-warning text-dark",
        "Mandatory HSSE Training"  => "bg-info",
        "Proton"                  => "bg-purple",
        _                         => "bg-secondary"
    };

    var statusBadgeClass = Model.GroupStatus switch
    {
        "Open"     => "bg-success",
        "Upcoming" => "bg-warning text-dark",
        "Closed"   => "bg-secondary",
        _          => "bg-secondary"
    };

    int completedPct = Model.TotalCount > 0
        ? (int)Math.Round(Model.CompletedCount * 100.0 / Model.TotalCount)
        : 0;

    int passRatePct = Model.CompletedCount > 0
        ? (int)Math.Round(Model.PassedCount * 100.0 / Model.CompletedCount)
        : 0;
}

<div class="container-fluid px-4 py-4">

    <!-- Back link -->
    <div class="mb-4">
        <a href="@ViewBag.BackUrl" class="btn btn-outline-secondary btn-sm">
            <i class="bi bi-arrow-left me-1"></i>Back to Monitoring
        </a>
    </div>

    <!-- Header -->
    <div class="mb-4">
        <div class="d-flex align-items-center gap-2 mb-1">
            <span class="badge @catBadgeClass">@Model.Category</span>
            <span class="badge @statusBadgeClass">@Model.GroupStatus</span>
        </div>
        <h2 class="fw-bold mb-1">@Model.Title</h2>
        <p class="text-muted mb-0">
            <i class="bi bi-calendar3 me-1"></i>@Model.Schedule.ToString("dd MMM yyyy")
        </p>
    </div>

    <!-- Summary cards -->
    <div class="row g-3 mb-4">
        <div class="col-sm-4">
            <div class="card border-0 shadow-sm text-center py-3">
                <div class="fs-3 fw-bold text-primary">@Model.TotalCount</div>
                <div class="text-muted small">Total Assigned</div>
            </div>
        </div>
        <div class="col-sm-4">
            <div class="card border-0 shadow-sm text-center py-3">
                <div class="fs-3 fw-bold text-success">@Model.CompletedCount</div>
                <div class="text-muted small">Completed (@completedPct%)</div>
            </div>
        </div>
        <div class="col-sm-4">
            <div class="card border-0 shadow-sm text-center py-3">
                @if (Model.CompletedCount == 0)
                {
                    <div class="fs-3 fw-bold text-muted">—</div>
                    <div class="text-muted small">Pass Rate</div>
                }
                else
                {
                    var passColor = passRatePct >= 70 ? "text-success" : "text-danger";
                    <div class="fs-3 fw-bold @passColor">@Model.PassedCount</div>
                    <div class="text-muted small">Passed (@passRatePct%)</div>
                }
            </div>
        </div>
    </div>

    <!-- Per-user table -->
    <div class="card border-0 shadow-sm">
        <div class="card-header bg-white py-3">
            <h5 class="mb-0 fw-semibold">
                <i class="bi bi-people me-2 text-primary"></i>Per-User Status
            </h5>
        </div>
        <div class="card-body p-0">
            <div class="table-responsive">
                <table class="table table-hover align-middle mb-0">
                    <thead class="table-light">
                        <tr>
                            <th>Name</th>
                            <th>NIP</th>
                            <th>Status</th>
                            <th>Score</th>
                            <th>Result</th>
                            <th>Completed At</th>
                        </tr>
                    </thead>
                    <tbody>
                        @if (Model.Sessions.Count == 0)
                        {
                            <tr>
                                <td colspan="6" class="text-center text-muted py-4">No sessions found for this group.</td>
                            </tr>
                        }
                        else
                        {
                            @foreach (var session in Model.Sessions)
                            {
                                var statusClass = session.UserStatus == "Completed" ? "bg-success" : "bg-light text-dark border";

                                string resultText;
                                string resultClass;
                                if (session.IsPassed == true)
                                {
                                    resultText  = "Pass";
                                    resultClass = "text-success fw-semibold";
                                }
                                else if (session.IsPassed == false)
                                {
                                    resultText  = "Fail";
                                    resultClass = "text-danger fw-semibold";
                                }
                                else
                                {
                                    resultText  = "—";
                                    resultClass = "text-muted";
                                }

                                <tr>
                                    <td class="fw-semibold">@session.UserFullName</td>
                                    <td class="text-muted small">@(string.IsNullOrEmpty(session.UserNIP) ? "—" : session.UserNIP)</td>
                                    <td><span class="badge @statusClass">@session.UserStatus</span></td>
                                    <td>
                                        @if (session.Score.HasValue)
                                        {
                                            <span>@session.Score%</span>
                                        }
                                        else
                                        {
                                            <span class="text-muted">—</span>
                                        }
                                    </td>
                                    <td class="@resultClass">@resultText</td>
                                    <td class="text-muted small">
                                        @(session.CompletedAt.HasValue ? session.CompletedAt.Value.ToString("dd MMM yyyy HH:mm") : "—")
                                    </td>
                                </tr>
                            }
                        }
                    </tbody>
                </table>
            </div>
        </div>
    </div>

</div>
```

Do not add `@section Scripts` — no JavaScript is needed on this read-only page.
