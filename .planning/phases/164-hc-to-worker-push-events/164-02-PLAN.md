    **4. Add SignalR event handlers** in `@section Scripts` block, AFTER assessment-hub.js loads:

    ```javascript
    // examClosed handler — reuses existing modal with dynamic text
    window.assessmentHub.on('examClosed', function(payload) {
        if (examClosed) return;  // dual-trigger guard
        examClosed = true;
        clearInterval(statusPollInterval);
        clearInterval(timerInterval);
        // Also clear auto-save interval (VERIFY variable name from code reading)
        window.onbeforeunload = null;

        var reasonText = (payload && payload.reason === 'hc_closed')
            ? 'Ujian diakhiri oleh pengawas.'
            : 'Waktu ujian habis.';
        document.getElementById('examClosedReason').textContent = reasonText;

        var modal = new bootstrap.Modal(document.getElementById('examClosedModal'));
        modal.show();
        // Wire countdown + redirect (reuse existing pattern from polling handler)
    });

    // sessionReset handler — new modal
    window.assessmentHub.on('sessionReset', function(payload) {
        // Stop all intervals
        clearInterval(timerInterval);
        clearInterval(statusPollInterval);
        // Clear auto-save interval (VERIFY variable name)
        window.onbeforeunload = null;

        // Disable form controls
        document.querySelectorAll('#examForm input, #examForm textarea, #examForm button, #examForm select').forEach(function(el) {
            el.disabled = true;
        });

        var modal = new bootstrap.Modal(document.getElementById('sessionResetModal'));
        modal.show();
    });

    // resetKembaliBtn click — redirect to Assessment list
    document.getElementById('resetKembaliBtn').addEventListener('click', function() {
        window.location.href = '@Url.Action("Assessment", "CMP")';
    });

    // Connection status badge
    var badge = document.getElementById('hubStatusBadge');
    window.assessmentHub.onreconnecting(function() {
        if (badge) { badge.className = 'badge bg-warning ms-2 small'; badge.textContent = 'Reconnecting...'; }
    });
    window.assessmentHub.onreconnected(function() {
        if (badge) { badge.className = 'badge bg-success ms-2 small'; badge.textContent = 'Live'; }
    });
    window.assessmentHub.onclose(function() {
        if (badge) { badge.className = 'badge bg-danger ms-2 small'; badge.textContent = 'Disconnected'; }
    });
    ```
