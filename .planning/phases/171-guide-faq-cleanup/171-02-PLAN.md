At the bottom of Guide.cshtml (in the `@section Scripts` block or before `</main>`), add JavaScript:
```javascript
function toggleAllFaq() {
    const btn = document.getElementById('faqToggleAll');
    const allCollapses = document.querySelectorAll('.faq-answer');
    const anyOpen = Array.from(allCollapses).some(el => el.classList.contains('show'));

    allCollapses.forEach(el => {
        const bsCollapse = bootstrap.Collapse.getOrCreateInstance(el, { toggle: false });
        if (anyOpen) {
            bsCollapse.hide();
        } else {
            bsCollapse.show();
        }
    });

    if (anyOpen) {
        btn.innerHTML = '<i class="bi bi-arrows-expand me-1"></i> Buka Semua';
    } else {
        btn.innerHTML = '<i class="bi bi-arrows-collapse me-1"></i> Tutup Semua';
    }
}
```
