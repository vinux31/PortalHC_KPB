```javascript
function updateExportHref() {
    const search = document.getElementById('searchInput')?.value ?? '';
    const section = document.getElementById('filterBagian')?.value ?? '';
    const unit = document.getElementById('filterUnit')?.value ?? '';
    const jalur = document.getElementById('filterJalur')?.value ?? '';
    const status = document.getElementById('filterStatus')?.value ?? '';
    const params = new URLSearchParams();
    if (search) params.set('search', search);
    if (section) params.set('section', section);
    if (unit) params.set('unit', unit);
    if (jalur) params.set('jalur', jalur);
    if (status) params.set('status', status);
    const base = '@Url.Action("ExportHistoriProton", "CDP")';
    const qs = params.toString();
    document.getElementById('btnExportHistoriProton').href = base + (qs ? '?' + qs : '');
}

// Attach to all filter elements
['searchInput','filterBagian','filterUnit','filterJalur','filterStatus'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.addEventListener('change', updateExportHref);
    if (el && el.tagName === 'INPUT') el.addEventListener('input', updateExportHref);
});
```
