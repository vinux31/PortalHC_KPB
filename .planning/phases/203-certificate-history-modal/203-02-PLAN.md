JS openHistoryModal function:
```javascript
function openHistoryModal(workerId, workerName, mode) {
    var modalBody = document.getElementById('history-modal-body');
    document.getElementById('history-modal-worker-name').textContent = workerName;
    modalBody.innerHTML = '<div class="text-center py-4"><div class="spinner-border text-primary" role="status"><span class="visually-hidden">Memuat...</span></div><p class="mt-2 text-muted">Memuat riwayat sertifikat...</p></div>';
    var modal = new bootstrap.Modal(document.getElementById('certificateHistoryModal'));
    modal.show();
    fetch('/Admin/CertificateHistory?workerId=' + encodeURIComponent(workerId) + '&mode=' + mode)
        .then(function(r) { if (!r.ok) throw new Error('HTTP ' + r.status); return r.text(); })
        .then(function(html) { modalBody.innerHTML = html; })
        .catch(function(e) {
            modalBody.innerHTML = '<div class="alert alert-danger mx-3 mt-3"><i class="bi bi-exclamation-triangle-fill me-2"></i>Gagal memuat riwayat sertifikat. Periksa koneksi lalu coba lagi.</div>';
        });
}
```

Event delegation JS:
```javascript
document.addEventListener('click', function(e) {
    var btn = e.target.closest('.btn-history');
    if (!btn) return;
    e.preventDefault();
    openHistoryModal(btn.dataset.workerId, btn.dataset.workerName, btn.dataset.mode || 'readonly');
});
```
