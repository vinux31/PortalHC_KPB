    .Where(t => !string.IsNullOrEmpty(t.Kategori) &&

3. Di JS function `filterTeamTable()` (sekitar line 266-323), ubah matchStatus logic. Ganti block matchStatus (line 287-294) menjadi:

```javascript
let matchStatus = true;
if (status === 'ALL') {
    matchStatus = true;
} else if (status === 'Sudah') {
    if (category) {
        const completedCats = row.getAttribute('data-completed-categories') || '';
        matchStatus = completedCats.includes(category);
    } else {
        matchStatus = rowHasTraining;
    }
} else if (status === 'Belum') {
    if (category) {
        const completedCats = row.getAttribute('data-completed-categories') || '';
        matchStatus = !completedCats.includes(category);
    } else {
        matchStatus = !rowHasTraining;
    }
}
```
