**A. AddTraining.cshtml** — Ganti dropdown Kategori hardcode (lines 90-103) dengan dynamic:

```html
<select id="kategoriSelect" asp-for="Kategori" class="form-select" required>
    <option value="">-- Pilih Kategori --</option>
    @foreach (var cat in (IEnumerable<dynamic>)ViewBag.KategoriOptions)
    {
        <option value="@cat.Name">@cat.Name</option>
    }
</select>
```

Tambah dropdown SubKategori di bawah Kategori (dalam form-group baru):
```html
<div class="mb-3">
    <label asp-for="SubKategori" class="form-label">Sub Kategori (opsional)</label>
    <select id="subKategoriSelect" asp-for="SubKategori" class="form-select" disabled>
        <option value="">-- Pilih Sub Kategori --</option>
        @foreach (var sub in (IEnumerable<dynamic>)ViewBag.SubKategoriOptions)
        {
            <option value="@sub.Name" data-parent-id="@sub.ParentId">@sub.Name</option>
        }
    </select>
</div>
```

Tambah JS di bagian scripts section (atau inline sebelum closing tag):
```javascript
<script>
(function() {
    const kategoriMap = {};
    @foreach (var cat in (IEnumerable<dynamic>)ViewBag.KategoriOptions)
    {
        @:kategoriMap["@cat.Name"] = @cat.Id;
    }

    const kategoriSelect = document.getElementById('kategoriSelect');
    const subKategoriSelect = document.getElementById('subKategoriSelect');

    kategoriSelect.addEventListener('change', function() {
        const selectedName = this.value;
        const parentId = kategoriMap[selectedName];
        const options = subKategoriSelect.querySelectorAll('option[data-parent-id]');

        options.forEach(opt => {
            opt.hidden = parentId === undefined || parseInt(opt.dataset.parentId) !== parentId;
        });

        subKategoriSelect.value = '';
        subKategoriSelect.disabled = !selectedName;
    });
})();
</script>
```
