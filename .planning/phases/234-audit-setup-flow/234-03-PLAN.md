        importResults.Add(new { RowNum = rowNum, Status = "Error", Message = "Kolom Deliverable kosong" });

    // D-15: Duplikasi detection — cek apakah sudah ada di DB (active)

    var existingDeliverable = await _context.ProtonDeliverables

        importResults.Add(new { RowNum = rowNum, Status = "Skipped", Message = "Sudah ada (aktif)" });
        continue;
    }

    validRows.Add(new { rowNum, bagian, unit, track, kompetensi, subKompetensi, deliverable, target });
    importResults.Add(new { RowNum = rowNum, Status = "Valid", Message = "OK" });

    return RedirectToAction("ImportSilabus");
}

using var transaction = await _context.Database.BeginTransactionAsync();

{
    if (newMappings.Any())
        await _context.CoachCoacheeMappings.AddRangeAsync(newMappings);
    // reactivated mappings sudah di-track oleh EF
    await _context.SaveChangesAsync();
    await transaction.CommitAsync();
}
