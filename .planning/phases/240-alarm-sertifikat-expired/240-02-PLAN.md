2. **Tambah method `TriggerCertExpiredNotificationsAsync`**:
   ```csharp
   private async Task TriggerCertExpiredNotificationsAsync()
   {
       try
       {
           // Query sertifikat expired (reuse logika serupa GetCertAlertCountsAsync tapi butuh detail)
           // TrainingRecords: expired, punya sertifikat, bukan Permanent, belum di-renew
           // AssessmentSessions: expired, punya sertifikat, belum di-renew
           // Untuk setiap record, butuh: Judul/Title dan NamaWorker/FullName

           var hcUsers = await _userManager.GetUsersInRoleAsync("HC");
           var adminUsers = await _userManager.GetUsersInRoleAsync("Admin");
           var targetUsers = hcUsers.Concat(adminUsers).DistinctBy(u => u.Id).ToList();

           // Pre-fetch existing CERT_EXPIRED notifications untuk performance
           var existingNotifications = await _context.UserNotifications
               .Where(n => n.Type == "CERT_EXPIRED")
               .Select(n => new { n.UserId, n.Message })
               .ToListAsync();

           foreach (var cert in expiredCerts)
           {
               var message = $"Sertifikat {cert.Judul} milik {cert.NamaWorker} telah expired";
               foreach (var targetUser in targetUsers)
               {
                   if (existingNotifications.Any(n => n.UserId == targetUser.Id && n.Message == message))
                       continue;

                   await _notificationService.SendAsync(
                       targetUser.Id,
                       "CERT_EXPIRED",
                       "Sertifikat Expired",
                       message,
                       "/Admin/RenewalCertificate"
                   );
               }
           }
       }
       catch (Exception ex)
       {
           _logger.LogWarning(ex, "Failed to trigger CERT_EXPIRED notifications");
       }
   }
   ```
