       var allowedExts = new[] { ".pdf", ".jpg", ".jpeg", ".png" };
       foreach (var wc in model.WorkerCerts!)

           if (wc.CertificateFile != null && wc.CertificateFile.Length > 0)

               var wcExt = Path.GetExtension(wc.CertificateFile.FileName).ToLowerInvariant();

       }
   }
