**4. Tambah JavaScript** — di bagian bawah sebelum existing scripts:

```javascript
@if (ViewBag.IsRenewalMode != true)
{
    <script>
    document.addEventListener('DOMContentLoaded', function() {
        var workerSelect = new TomSelect('#WorkerSelect', {
            maxItems: 20,
            plugins: ['remove_button'],
            placeholder: 'Pilih pekerja (maks 20)...',
            onItemAdd: function(value) {
                var option = this.options[value];
                addWorkerRow(value, option.text);
                updateWorkerCount();
                updateSubmitButton();
            },
            onItemRemove: function(value) {
                removeWorkerRow(value);
                reindexWorkerRows();
                updateWorkerCount();
                updateSubmitButton();
            }
        });

        var submitBtn = document.querySelector('button[type="submit"]');
        if (submitBtn) submitBtn.disabled = true;

        function addWorkerRow(userId, userName) {
            var container = document.getElementById('workerCertsContainer');
            var emptyMsg = document.getElementById('emptyWorkerMsg');
            if (emptyMsg) emptyMsg.style.display = 'none';

            var index = container.querySelectorAll('.worker-cert-card').length;
            var row = document.createElement('div');
            row.className = 'card mb-2 worker-cert-card';
            row.dataset.userId = userId;
            row.innerHTML =
                '<div class="card-body py-2">' +
                    '<div class="fw-bold mb-2">' + escapeHtml(userName) + '</div>' +
                    '<input type="hidden" name="WorkerCerts[' + index + '].UserId" value="' + userId + '" />' +
                    '<div class="row g-2">' +
                        '<div class="col-md-6">' +
                            '<label class="form-label">File Sertifikat</label>' +
                            '<input type="file" name="WorkerCerts[' + index + '].CertificateFile" class="form-control" accept=".pdf,.jpg,.jpeg,.png" />' +
                            '<div class="form-text">PDF, JPG, PNG. Maks 10MB.</div>' +
                        '</div>' +
                        '<div class="col-md-6">' +
                            '<label class="form-label">Nomor Sertifikat</label>' +
                            '<input type="text" name="WorkerCerts[' + index + '].NomorSertifikat" class="form-control" placeholder="Nomor sertifikat" />' +
                        '</div>' +
                    '</div>' +
                '</div>';
            container.appendChild(row);
        }

        function removeWorkerRow(userId) {
            var container = document.getElementById('workerCertsContainer');
            var card = container.querySelector('.worker-cert-card[data-user-id="' + userId + '"]');
            if (card) card.remove();

            var remaining = container.querySelectorAll('.worker-cert-card');
            if (remaining.length === 0) {
                var emptyMsg = document.getElementById('emptyWorkerMsg');
                if (emptyMsg) emptyMsg.style.display = '';
            }
        }

        function reindexWorkerRows() {
            var container = document.getElementById('workerCertsContainer');
            var cards = container.querySelectorAll('.worker-cert-card');
            cards.forEach(function(card, i) {
                var hidden = card.querySelector('input[type="hidden"]');
                if (hidden) hidden.name = 'WorkerCerts[' + i + '].UserId';
                var fileInput = card.querySelector('input[type="file"]');
                if (fileInput) fileInput.name = 'WorkerCerts[' + i + '].CertificateFile';
                var textInput = card.querySelector('input[type="text"]');
                if (textInput) textInput.name = 'WorkerCerts[' + i + '].NomorSertifikat';
            });
        }

        function updateWorkerCount() {
            var badge = document.getElementById('workerCountBadge');
            var count = document.querySelectorAll('.worker-cert-card').length;
            if (count > 0) {
                badge.textContent = count + ' pekerja';
                badge.style.display = '';
            } else {
                badge.style.display = 'none';
            }
        }

        function updateSubmitButton() {
            var submitBtn = document.querySelector('button[type="submit"]');
            var count = document.querySelectorAll('.worker-cert-card').length;
            if (submitBtn) submitBtn.disabled = (count === 0);
        }

        function escapeHtml(text) {
            var div = document.createElement('div');
            div.appendChild(document.createTextNode(text));
            return div.innerHTML;
        }

        // Submit loading state
        var form = document.querySelector('form');
        if (form) {
            form.addEventListener('submit', function() {
                var btn = form.querySelector('button[type="submit"]');
                if (btn && !btn.disabled) {
                    btn.disabled = true;
                    btn.innerHTML = '<span class="spinner-border spinner-border-sm me-2"></span>Menyimpan...';
                }
            });
        }
    });
    </script>
}
```
