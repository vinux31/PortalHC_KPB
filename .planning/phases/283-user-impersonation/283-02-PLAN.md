   f. **JavaScript autocomplete** (per D-03) — di bagian scripts _Layout:
   ```javascript
   // Impersonation user search autocomplete
   (function() {
       var searchInput = document.getElementById('impersonate-search');
       if (!searchInput) return;
       var resultsDiv = document.getElementById('impersonate-results');
       var debounceTimer;

       searchInput.addEventListener('input', function() {
           clearTimeout(debounceTimer);
           var q = this.value.trim();
           if (q.length < 2) { resultsDiv.innerHTML = ''; return; }
           debounceTimer = setTimeout(function() {
               fetch('/Admin/SearchUsersApi?q=' + encodeURIComponent(q))
                   .then(function(r) { return r.json(); })
                   .then(function(users) {
                       if (users.length === 0) {
                           resultsDiv.innerHTML = '<div class="list-group-item text-muted fst-italic small">Tidak ada user ditemukan — coba kata kunci atau NIP lain</div>';
                           return;
                       }
                       resultsDiv.innerHTML = users.map(function(u) {
                           return '<button type="button" class="list-group-item list-group-item-action py-1 px-2" data-userid="' + u.id + '">'
                               + '<strong>' + u.fullName + '</strong> <small class="text-muted">' + u.nip + '</small>'
                               + ' <span class="badge bg-secondary">' + u.selectedView + '</span></button>';
                       }).join('');
                   });
           }, 300);
       });

       resultsDiv.addEventListener('click', function(e) {
           var btn = e.target.closest('[data-userid]');
           if (!btn) return;
           // Create and submit form
           var form = document.createElement('form');
           form.method = 'POST';
           form.action = '/Admin/StartImpersonation';
           form.innerHTML = '<input name="mode" value="user"><input name="targetUserId" value="' + btn.dataset.userid + '">'
               + '<input name="__RequestVerificationToken" value="' + document.querySelector('input[name=__RequestVerificationToken]').value + '">';
           document.body.appendChild(form);
           form.submit();
       });
   })();
   ```

   g. **Read-only JS** (per D-05) — disable submit buttons saat impersonation:
   ```javascript
   // Read-only mode: disable submit buttons during impersonation
   if (document.body.dataset.impersonating === 'true') {
       document.querySelectorAll('button[type=submit], input[type=submit]').forEach(function(btn) {
           // Skip StopImpersonation button
           if (btn.closest('form[action*="StopImpersonation"]')) return;
           btn.disabled = true;
           btn.style.opacity = '0.6';
           var badge = document.createElement('span');
           badge.className = 'badge bg-danger ms-2';
           badge.textContent = 'Mode Read-Only';
           btn.parentNode.insertBefore(badge, btn.nextSibling);
       });
   }
   ```

   h. **AJAX 403 toast** — intercept fetch 403 responses:
   ```javascript
   // Intercept AJAX 403 for read-only toast
   if (document.body.dataset.impersonating === 'true') {
       var origFetch = window.fetch;
       window.fetch = function() {
           return origFetch.apply(this, arguments).then(function(response) {
               if (response.status === 403) {
                   response.clone().json().then(function(data) {
                       if (data.readOnly) {
                           // Show Bootstrap toast
                           var toastHtml = '<div class="toast align-items-center text-bg-danger border-0 show" role="alert" style="position:fixed;top:60px;right:20px;z-index:1050;">'
                               + '<div class="d-flex"><div class="toast-body"><i class="bi bi-shield-lock me-2"></i>Aksi ini diblokir — Mode Read-Only aktif</div>'
                               + '<button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast"></button></div></div>';
                           var container = document.createElement('div');
                           container.innerHTML = toastHtml;
                           document.body.appendChild(container.firstChild);
                           setTimeout(function() { container.firstChild && container.firstChild.remove(); }, 4000);
                       }
                   }).catch(function(){});
               }
               return response;
           });
       };
   }
   ```
