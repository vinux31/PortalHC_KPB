   ```javascript
   document.getElementById('QuestionType').addEventListener('change', function() {
       const qtype = this.value;
       const optionsSection = document.getElementById('optionsSection');
       const rubrikSection = document.getElementById('rubrikSection');
       const scoreInput = document.getElementById('ScoreValue');

       optionsSection.style.display = (qtype === 'Essay') ? 'none' : '';
       rubrikSection.style.display = (qtype === 'Essay') ? '' : 'none';
       scoreInput.disabled = (qtype !== 'Essay'); // MC/MA fixed 10
       if (qtype !== 'Essay') scoreInput.value = 10;

       // Ganti radio → checkbox untuk MA (per D-02, D-05)
       document.querySelectorAll('.correct-input').forEach(inp => {
           inp.type = (qtype === 'MultipleAnswer') ? 'checkbox' : 'radio';
           if (qtype !== 'MultipleAnswer') inp.checked = false; // reset
       });
   });
   ```
