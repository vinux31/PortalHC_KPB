**5. JavaScript auto-save:**

MA auto-save (per D-21 — immediate, no debounce):
```javascript
document.querySelectorAll('.exam-checkbox').forEach(cb => {
    cb.addEventListener('change', function() {
        const qId = this.dataset.questionId;
        const selected = [...document.querySelectorAll(`input.exam-checkbox[data-question-id="${qId}"]:checked`)]
            .map(el => el.value);
        connection.invoke("SaveMultipleAnswer", sessionId, parseInt(qId), selected.join(','));
        // Update hidden input ans_N for form submit fallback
        document.getElementById('ans_' + qId).value = selected.join(',');
        // Update card background (D-19: bg-success bg-opacity-10)
    });
});
```

Essay auto-save (per D-21 — debounce 2 detik):
```javascript
let essayTimers = {};
document.querySelectorAll('.exam-essay').forEach(ta => {
    ta.addEventListener('input', function() {
        const qId = this.dataset.questionId;
        const charCount = this.value.length;
        const maxChar = parseInt(this.getAttribute('maxlength'));
        // Update counter (D-22)
        const counter = document.getElementById('charCount_' + qId);
        counter.textContent = charCount;
        counter.className = charCount > maxChar * 0.9 ? 'text-warning' : (charCount >= maxChar ? 'text-danger' : 'text-muted');

        clearTimeout(essayTimers[qId]);
        essayTimers[qId] = setTimeout(() => {
            connection.invoke("SaveTextAnswer", sessionId, parseInt(qId), this.value);
        }, 2000);
    });
});
```
