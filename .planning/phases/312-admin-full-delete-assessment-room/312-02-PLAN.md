**C. Append `<script>` block** dengan modal handler (event delegation di `document.body` untuk HTMX safety per Pitfall 5):

```html
<script>
(function () {
    // Q2 RESOLVED: handler attached via event delegation di document.body (stable across DOM mutations).
    // Idempotent flag mencegah double-bind. htmx:afterSwap listener reset flag + re-init kalau partial swap menghapus script execution context.
    if (window.__dam312Attached) {
        // Sudah ter-attach sebelumnya — handler di document.body masih aktif (delegation), no re-bind needed.
        // Tapi kita tetap pastikan flag stay true (idempotent).
        return;
    }
    window.__dam312Attached = true;

    var appUrl = function (path) {
        var base = (document.querySelector('base')?.getAttribute('href') || '/').replace(/\/$/, '');
        return base + path;
    };

    // Reset modal state ke Step 1 + loading
    function resetModal() {
        document.getElementById('dam-step-1').style.display = '';
        document.getElementById('dam-step-2').style.display = 'none';
        document.getElementById('dam-footer-1').style.display = '';
        document.getElementById('dam-footer-2').style.display = 'none';
        document.getElementById('dam-loading').style.display = '';
        document.getElementById('dam-loading').innerHTML = '<span class="spinner-border spinner-border-sm me-2" aria-hidden="true"></span>Memuat dampak penghapusan...';
        document.getElementById('dam-content').style.display = 'none';
        document.getElementById('dam-prepost-breakdown').style.display = 'none';
        document.getElementById('dam-next-btn').disabled = true;
        document.getElementById('dam-dialog').classList.remove('modal-lg');
    }

    // show.bs.modal: read data-* attributes, fetch impact, populate Step 1
    document.body.addEventListener('show.bs.modal', function (event) {
        var modalEl = event.target;
        if (!modalEl || modalEl.id !== 'deleteAssessmentModal') return;

        var trigger = event.relatedTarget;
        if (!trigger) return;

        var type = trigger.dataset.deleteType;       // "single" | "group" | "prepost"
        var id   = trigger.dataset.deleteId;
        var title = trigger.dataset.deleteTitle || '';

        resetModal();

        // Set title per type (per UI-SPEC line 106-110)
        var titleMap = {
            'single':  'Hapus Assessment: ' + title,
            'group':   'Hapus Grup Assessment: ' + title,
            'prepost': 'Hapus Grup Pre-Post Test: ' + title
        };
        document.getElementById('dam-title').textContent = titleMap[type] || ('Hapus: ' + title);

        // Cascade item 1 per type (per UI-SPEC line 134-136)
        var cascadeMap = {
            'single':  'Sesi assessment beserta semua paket soal, soal, dan opsi',
            'group':   'Semua sesi assessment dalam grup beserta paket soal, soal, dan opsi',
            'prepost': 'Kedua sesi (Pre-Test dan Post-Test) beserta paket soal, soal, dan opsi'
        };
        document.getElementById('dam-cascade-1').textContent = cascadeMap[type] || cascadeMap['single'];

        // Form action + hidden field setup
        var form = document.getElementById('dam-submit-form');
        var formId = document.getElementById('dam-form-id');
        var formLinkedId = document.getElementById('dam-form-linkedid');
        formId.disabled = true;
        formLinkedId.disabled = true;

        if (type === 'single') {
            form.action = appUrl('/Admin/DeleteAssessment');
            formId.value = id;
            formId.disabled = false;
        } else if (type === 'group') {
            form.action = appUrl('/Admin/DeleteAssessmentGroup');
            formId.value = id;
            formId.disabled = false;
        } else if (type === 'prepost') {
            form.action = appUrl('/Admin/DeletePrePostGroup');
            formLinkedId.value = id;
            formLinkedId.disabled = false;
            document.getElementById('dam-dialog').classList.add('modal-lg');
        }

        // Fetch impact
        var url = appUrl('/Admin/GetDeleteImpact')
            + '?type=' + encodeURIComponent(type)
            + '&id=' + encodeURIComponent(id);

        fetch(url, { credentials: 'same-origin' })
            .then(function (r) {
                if (!r.ok) throw new Error('HTTP ' + r.status);
                return r.json();
            })
            .then(function (data) {
                document.getElementById('dam-status').textContent = data.status || '-';
                document.getElementById('dam-response-count').textContent = data.responseCount || 0;
                document.getElementById('dam-cert-count').textContent = data.certCount || 0;
                document.getElementById('dam-package-count').textContent = data.packageCount || 0;
                document.getElementById('dam-attempt-count').textContent = data.attemptCount || 0;

                if (type === 'prepost' && data.prePostBreakdown) {
                    document.getElementById('dam-pre-status').textContent = data.prePostBreakdown.pre.status || '-';
                    document.getElementById('dam-pre-resp').textContent = data.prePostBreakdown.pre.responseCount || 0;
                    document.getElementById('dam-post-status').textContent = data.prePostBreakdown.post.status || '-';
                    document.getElementById('dam-post-resp').textContent = data.prePostBreakdown.post.responseCount || 0;
                    document.getElementById('dam-prepost-breakdown').style.display = '';
                }

                document.getElementById('dam-loading').style.display = 'none';
                document.getElementById('dam-content').style.display = '';
                document.getElementById('dam-next-btn').disabled = false;
            })
            .catch(function (err) {
                document.getElementById('dam-loading').innerHTML =
                    '<span class="text-danger"><i class="bi bi-exclamation-circle me-1"></i>Gagal memuat data dampak. Tutup dan coba lagi.</span>';
                // Lanjutkan tetap disabled
            });
    });

    // Step navigation: Lanjutkan → Step 2
    document.body.addEventListener('click', function (event) {
        var t = event.target.closest('#dam-next-btn');
        if (t && !t.disabled) {
            document.getElementById('dam-step-1').style.display = 'none';
            document.getElementById('dam-step-2').style.display = '';
            document.getElementById('dam-footer-1').style.display = 'none';
            document.getElementById('dam-footer-2').style.display = '';
            document.getElementById('dam-back-btn').focus();
            return;
        }
        var b = event.target.closest('#dam-back-btn');
        if (b) {
            document.getElementById('dam-step-1').style.display = '';
            document.getElementById('dam-step-2').style.display = 'none';
            document.getElementById('dam-footer-1').style.display = '';
            document.getElementById('dam-footer-2').style.display = 'none';
            document.getElementById('dam-next-btn').focus();
            return;
        }
    });

    // Q2 RESOLVED: htmx:afterSwap re-init.
    // Handler di document.body adalah event-delegated (stable), TAPI script <script> di partial bisa di-re-evaluate
    // saat HTMX swap partial. Listener ini idempotent guard: kalau swap kebetulan re-load script tag,
    // flag __dam312Attached sudah true → skip re-bind. Kalau ada edge case dimana DOM modal di-replace
    // (modal markup di partial yg di-swap), kita pastikan element references tetap fresh via selector lookup
    // pada show.bs.modal event (handler tidak cache element references — selalu document.getElementById).
    document.body.addEventListener('htmx:afterSwap', function (event) {
        // Defensive: kalau modal element hilang setelah swap, log warning ke console.
        // Modal SHOULD persist di partial bottom (per Q2 RESOLVED — placement di partial OK karena
        // event delegation di document.body bukan modalEl.addEventListener).
        if (!document.getElementById('deleteAssessmentModal')) {
            console.warn('[Phase 312] deleteAssessmentModal hilang setelah htmx:afterSwap — verifikasi partial render');
        }
        // No re-bind needed: handlers di document.body (event delegation) survive DOM mutations.
    });
})();
</script>
```
