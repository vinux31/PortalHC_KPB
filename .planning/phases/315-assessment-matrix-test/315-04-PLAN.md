 *
 * Source-of-truth design spec: docs/superpowers/specs/2026-05-11-assessment-matrix-test-design.md (commit 94bacecf)
 *
 * Output:
 *   - docs/test-reports/YYYY-MM-DD-assessment-matrix.md (generated by globalTeardown)
 *   - test-results/matrix-s{N}-{step}.png (screenshots per finding)

import { test, expect, Browser } from '@playwright/test';
import { readFileSync } from 'fs';
import type { ScenarioConfig } from './helpers/matrixTypes';
import { takeExam, gradeEssaysAsHc, verifyResultPage } from './helpers/examMatrix';
import { collector, softAssert, SkipScenarioError } from './helpers/matrixReport';

// Serial mode — singleton collector + sequential scenario execution (Pattern 3 caveat)
test.describe.configure({ mode: 'serial' });

// Read state file written by global.setup.ts
const stateRaw = readFileSync('tests/.matrix-state.json', 'utf-8');
const state: { snapshotPath: string; seededAt: string; scenarios: ScenarioConfig[] } = JSON.parse(stateRaw);

// Helper: orchestrate 1 discovery scenario (peserta1 + peserta2 + HC grading + verify)
async function runDiscoveryScenario(browser: Browser, cfg: ScenarioConfig) {
  const ctx1 = await browser.newContext();
  const page1 = await ctx1.newPage();
  const ctx2 = await browser.newContext();
  const page2 = await ctx2.newPage();
  const ctxHc = await browser.newContext();
  const pageHc = await ctxHc.newPage();

  try {
    // Peserta 1 — take exam normally
    await takeExam(page1, cfg, 'coachee', cfg.sessionIdPeserta1);

    // Peserta 2 — take exam dengan sabotage 1 question (deterministic — first Q)
    await takeExam(page2, cfg, 'coachee2', cfg.sessionIdPeserta2, { sabotageOneAnswer: true });

    // HC — grade essays jika ada
    if (cfg.hasEssay) {
      await gradeEssaysAsHc(pageHc, cfg);
    }

    // Peserta 1 — verify result page menampilkan score
    await verifyResultPage(page1, cfg, 'coachee');
    await verifyResultPage(page2, cfg, 'coachee2');
  } catch (e) {
    if (e instanceof SkipScenarioError) {
      console.log(`[Scenario S${cfg.id}] Critical fail at step — skip sisa scenario:`, e.message);
      return;  // softAssert sudah record finding; lanjut scenario berikut
    }
    throw e;  // unexpected — let Playwright handle
  } finally {
    await ctx1.close();
    await ctx2.close();
    await ctxHc.close();
  }
}

// =====================================================================
// Group 1: Matrix Discovery (mixed by assessment type) — S1-S4
// =====================================================================
test.describe('Matrix - Discovery (mixed by assessment type)', () => {
  for (const cfg of state.scenarios.filter(s => s.id >= 1 && s.id <= 4)) {
    test(`S${cfg.id} ${cfg.title}`, async ({ browser }) => {
      await runDiscoveryScenario(browser, cfg);
    });
  }
});

// =====================================================================
// Group 2: Matrix Discovery (single-type Online) — S5-S7
// =====================================================================
test.describe('Matrix - Discovery (single-type Online)', () => {
  for (const cfg of state.scenarios.filter(s => s.id >= 5 && s.id <= 7)) {
    test(`S${cfg.id} ${cfg.title}`, async ({ browser }) => {
      await runDiscoveryScenario(browser, cfg);
    });
  }
});

// =====================================================================
// Group 3: Meta-validation sentinels — S8-S10
// =====================================================================
test.describe('Matrix - Meta-validation sentinels', () => {

  // S8 [META-AllCorrect]: peserta1 + peserta2 SEMUA benar (no sabotage) — expect 100% score
  test('S8 [META-AllCorrect] Sentinel', async ({ browser }) => {
    const cfg = state.scenarios.find(s => s.id === 8);
    if (!cfg) throw new Error('Scenario 8 not found in state');
    cfg.title = `[META-AllCorrect] ${cfg.title}`;  // tag for filter

    const ctx1 = await browser.newContext();
    const page1 = await ctx1.newPage();
    const ctxHc = await browser.newContext();
    const pageHc = await ctxHc.newPage();
    try {
      await takeExam(page1, cfg, 'coachee', cfg.sessionIdPeserta1);  // tidak ada sabotage
      if (cfg.hasEssay) await gradeEssaysAsHc(pageHc, cfg);
      // Sentinel: expected 100% — kalau bukan, record meta finding
      await softAssert(
        { scenario: cfg, step: 'sentinel-all-correct-score', severity: 'minor', page: page1, isMeta: true },
        async () => {
          await page1.goto(`/CMP/Results/${cfg.sessionIdPeserta1}`);
          const scoreText = await page1.locator('.score-badge, [data-score], .result-score').first().textContent({ timeout: 10_000 });
          if (!scoreText?.match(/100|full|sempurna/i)) {
            throw new Error(`Expected 100% / sempurna score, got: "${scoreText}"`);
          }
        },
        'Sentinel AllCorrect: score = 100%'
      );
    } finally {
      await ctx1.close();
      await ctxHc.close();
    }
  });

  // S9 [META-AllWrong]: peserta1 sabotage ALL answers — expect 0% score
  test('S9 [META-AllWrong] Sentinel', async ({ browser }) => {
    const cfg = state.scenarios.find(s => s.id === 9);
    if (!cfg) throw new Error('Scenario 9 not found in state');
    cfg.title = `[META-AllWrong] ${cfg.title}`;

    const ctx1 = await browser.newContext();
    const page1 = await ctx1.newPage();
    try {
      // Sabotage: gunakan modified takeExam dengan all-wrong mode (atau loop manual)
      // Implementasi: pakai sabotageOneAnswer=true (limited) ATAU buat copy questions dengan correctOptionIds dikosongkan
      // Pendekatan terjamin: clone cfg dengan questions[].correctOptionIds = []
      const cfgAllWrong = {
        ...cfg,
        questions: cfg.questions.map(q => ({ ...q, correctOptionIds: [] })),
      };
      await takeExam(page1, cfgAllWrong, 'coachee', cfg.sessionIdPeserta1);

      await softAssert(
        { scenario: cfg, step: 'sentinel-all-wrong-score', severity: 'minor', page: page1, isMeta: true },
        async () => {
          await page1.goto(`/CMP/Results/${cfg.sessionIdPeserta1}`);
          const scoreText = await page1.locator('.score-badge, [data-score], .result-score').first().textContent({ timeout: 10_000 });
          if (!scoreText?.match(/0|nol/i)) {
            throw new Error(`Expected 0% / nol score, got: "${scoreText}"`);
          }
        },
        'Sentinel AllWrong: score = 0%'
      );
    } finally {
      await ctx1.close();
    }
  });

  // S10 [META-CollectorCheck]: SENGAJA gagal supaya verifikasi collector working
  // D-06: pakai test.fail() — exit 0 saat fail expected, exit non-zero kalau pass (collector rusak)
  test.fail('S10 [META-CollectorCheck] Sentinel — sengaja gagal untuk verifikasi collector', async ({ browser }) => {
    const cfg = state.scenarios.find(s => s.id === 10);
    if (!cfg) throw new Error('Scenario 10 not found in state');
    cfg.title = `[META-CollectorCheck] ${cfg.title}`;

    const ctx1 = await browser.newContext();
    const page1 = await ctx1.newPage();
    try {
      // Force critical softAssert fail — collector akan record finding + throw SkipScenarioError
      await softAssert(
        { scenario: cfg, step: 'sentinel-collector-check', severity: 'critical', page: page1, isMeta: true },
        async () => {
          throw new Error('Sentinel intentionally fails — verify collector records this finding');
        },
        'Sentinel CollectorCheck: collector records meta finding'
      );
    } finally {
      await ctx1.close();
    }
  });
});

**Catatan implementasi:**

1. **Top-level `state` read** dilakukan di module scope (di luar test.describe). Jika `.matrix-state.json` belum ditulis (e.g. user run spec tanpa global setup) → `readFileSync` throw → entire file fail import → user dapat clear error message. Acceptable behavior.

2. **Dynamic test generation:** loop `state.scenarios.filter(...)` di dalam `test.describe()` body. Test names unik (`S${cfg.id} ${cfg.title}`) — `--grep "Scenario 5"` ATAU `--grep "S5"` akan match scenario 5.

3. **test.fail() di S10:** Playwright treat test sebagai expected failure. Exit code 0 kalau body throw (collector record), exit non-zero kalau body pass tanpa throw (collector broken).

4. **Browser context cleanup:** WAJIB `finally { ctx.close(); }` untuk avoid resource leak across sequential scenarios.

5. **SkipScenarioError handling:** catch di `runDiscoveryScenario`, return tanpa rethrow. Finding sudah ter-record di collector.
