```typescript
function renderFinding(f: Finding): string {
  return [
    `### Scenario ${f.scenarioId}: ${f.scenarioTitle} — ${f.step}`,
    ``,
    `- **Severity:** ${f.severity}`,
    `- **Expected:** ${f.expected}`,
    `- **Actual:** ${f.actual}`,
    f.screenshotPath ? `- **Screenshot:** \`${f.screenshotPath}\`` : '',
    `- **Hypothesis:** ${deriveHypothesis(f)}`,
  ].filter(Boolean).join('\n');
}

function deriveHypothesis(f: Finding): string {
  const step = f.step.toLowerCase();
  const actual = (f.actual || '').toLowerCase();

  if (step.includes('signalr-ready')) {
    return 'Hub belum `Connected` dalam 10s. Cek SignalR negotiate roundtrip; verifikasi `Hubs/AssessmentHub.cs OnConnectedAsync` + browser DevTools Network tab (filter `assessmentHub/negotiate`).';
  }
  if (step.includes('mc-q') && (actual.includes('timeout') || actual.includes('saved'))) {
    return 'MC HTTP `/CMP/SaveAnswer` mungkin 500/timeout, atau `#saveIndicatorText` selector berubah. Cek `Controllers/CMPController.cs:348-417` (SaveAnswer) + Network tab response status.';
  }
  if (step.includes('ma-q') && (actual.includes('timeout') || actual.includes('saved'))) {
    return 'MA `SaveMultipleAnswer` SignalR hub invoke mungkin gagal. Cek `Hubs/AssessmentHub.cs:188-252` + hub state log + verifikasi `correctOptionIds` di seed match dengan DOM `data-question-id` + `value`.';
  }
  if (step.includes('essay-q') && (actual.includes('timeout') || actual.includes('saved'))) {
    return 'Essay `SaveTextAnswer` SignalR mungkin gagal atau debounce <2s mismatched. Cek `Hubs/AssessmentHub.cs:134-182` + `Views/CMP/StartExam.cshtml:861-904` (2s setTimeout) + textarea event listener.';
  }
  if (step.includes('submit-exam') && actual.includes('results')) {
    return 'SubmitExam mungkin redirect ke `/CMP/ExamSummary/{id}` (timer expired) atau render alert banner. Cek `Controllers/CMPController.cs:1569+` SubmitExam + verify timer state pre-submit.';
  }
  if (step.includes('verify-result-page')) {
    return 'Score badge selector tidak ketemu — mungkin scoring belum komplet (Essay grading pending) atau selector view berubah. Cek `Views/CMP/Results.cshtml` + verifikasi `gradeEssaysAsHc` step lulus + `AssessmentSession.Status = "Completed"` di DB.';
  }
  if (step.includes('hc-grade-essays')) {
    return 'HC essay grading workflow gagal — selector input score atau finalize button mungkin berubah. Cek `Controllers/AssessmentAdminController.cs:2873-2950` (SubmitEssayScore + FinalizeEssayGrading) + AssessmentMonitoringDetail view.';
  }
  if (step.includes('navigate-start-exam')) {
    return 'StartExam page tidak render `#examForm` — kemungkinan: (1) sessionId tidak owned by login user, (2) UserPackageAssignment belum ter-create, (3) view error. Cek `Controllers/CMPController.cs:880-1000` (StartExam) + AssessmentSession.Status valid.';
  }
  if (step.includes('hc-navigate-monitoring-detail')) {
    return 'AssessmentMonitoringDetail page tidak menampilkan title — kemungkinan URL encoding masalah, atau query string binding gagal. Cek `Controllers/AssessmentAdminController.cs:2684-2702` (AssessmentMonitoringDetail) + verifikasi URL via DevTools Network tab.';
  }

  return 'Hypothesis otomatis tidak tersedia — periksa screenshot, URL bar, browser console log, dan Playwright trace (kalau di-enable).';
}
```
