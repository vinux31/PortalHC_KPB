    test.describe('FLOW U — ManageCategories CRUD + Duplicate Reject', () => {
      let catName: string;
      let editedName: string;
      const escapeSql = (s: string) => s.replace(/'/g, "''");

      test('U1 — HC create category + DB verify INSERT', async ({ page }) => {
        test.setTimeout(FLOW_TIMEOUT_MS);
        catName = `[319-U] OJT-${Date.now()}`;
        await login(page, 'hc');
        await page.goto('/AssessmentAdmin/ManageCategories');
        await page.waitForLoadState('networkidle');

        // Expand add form collapse
        const expandBtn = page.locator('button[data-bs-target="#addCategoryForm"]').first();
        await expandBtn.click();
        await expect(page.locator('#addCategoryForm.show')).toBeVisible({ timeout: 5_000 });

        // Fill form fields
        await page.fill('#addCategoryForm input[name="name"]', catName);
        const passInput = page.locator('#addCategoryForm input[name="defaultPassPercentage"]');
        if (await passInput.count() > 0) await passInput.fill('75');
        const sortInput = page.locator('#addCategoryForm input[name="sortOrder"]');
        if (await sortInput.count() > 0) await sortInput.fill('99');

        // Submit + wait redirect (POST-redirect-GET pattern)
        const submitBtn = page.locator('#addCategoryForm button[type="submit"]').filter({ hasText: /tambah|simpan|save/i }).first();
        await Promise.all([
          page.waitForURL(/\/ManageCategories/, { timeout: 10_000 }),
          submitBtn.click(),
        ]);

        // Assert success alert + table contains row
        await expect(
          page.locator('.alert-success', { hasText: /berhasil/i })
        ).toBeVisible({ timeout: 10_000 });
        await expect(page.locator('td', { hasText: catName }).first()).toBeVisible({ timeout: 5_000 });

        // DB verify
        const count = await db.queryScalar(
          `SELECT COUNT(*) FROM AssessmentCategories WHERE Name = N'${escapeSql(catName)}'`
        );
        expect(count).toBe(1);
      });

      test('U2 — HC edit category name + DB verify UPDATE', async ({ page }) => {
        test.setTimeout(FLOW_TIMEOUT_MS);
        editedName = `${catName} EDITED`;
        await login(page, 'hc');
        await page.goto('/AssessmentAdmin/ManageCategories');
        await page.waitForLoadState('networkidle');

        // Find row by catName → click edit button (data-bs-toggle modal atau inline edit form)
        const row = page.locator('tr', { hasText: catName }).first();
        await expect(row).toBeVisible({ timeout: 5_000 });

        // Try edit button via data-bs-target modal pattern; fallback ke anchor with /EditCategory in href
        const editBtn = row.locator('button[data-bs-target*="edit"], a[href*="EditCategory"], button:has-text("Edit")').first();
        await editBtn.click();
        await page.waitForLoadState('networkidle');

        // Edit form input — could be in modal atau inline; find input with current name
        const nameInput = page.locator('input[name="name"]').filter({ has: page.locator('xpath=.[@value="' + catName.replace(/"/g, '\\"') + '"]') }).first();
        const fallbackInput = page.locator('input[name="name"]').first();
        const targetInput = (await nameInput.count() > 0) ? nameInput : fallbackInput;
        await targetInput.fill(editedName);

        // Submit edit form
        const editSubmit = page.locator('button[type="submit"]').filter({ hasText: /simpan|update|save/i }).first();
        await Promise.all([
          page.waitForURL(/\/ManageCategories/, { timeout: 10_000 }),
          editSubmit.click(),
        ]);

        // DB verify
        const newCount = await db.queryScalar(
          `SELECT COUNT(*) FROM AssessmentCategories WHERE Name = N'${escapeSql(editedName)}'`
        );
        expect(newCount).toBe(1);
        const oldCount = await db.queryScalar(
          `SELECT COUNT(*) FROM AssessmentCategories WHERE Name = N'${escapeSql(catName)}'`
        );
        expect(oldCount).toBe(0);
      });

      test('U3 — HC delete category + DB verify row removed', async ({ page }) => {
        test.setTimeout(FLOW_TIMEOUT_MS);
        await login(page, 'hc');
        await page.goto('/AssessmentAdmin/ManageCategories');
        await page.waitForLoadState('networkidle');

        // Find row by editedName → delete button
        const row = page.locator('tr', { hasText: editedName }).first();
        await expect(row).toBeVisible({ timeout: 5_000 });

        // Try delete button — accept browser confirm dialog
        page.once('dialog', d => d.accept());
        const deleteBtn = row.locator('button:has-text("Hapus"), button:has-text("Delete"), a[href*="DeleteCategory"], button[data-bs-target*="delete"]').first();
        await deleteBtn.click();

        // If modal opens, click confirm inside modal
        const modalConfirm = page.locator('.modal.show button[type="submit"], .modal.show button:has-text("Hapus"), .modal.show button:has-text("Ya")').first();
        if (await modalConfirm.count() > 0 && await modalConfirm.isVisible()) {
          await Promise.all([
            page.waitForURL(/\/ManageCategories/, { timeout: 10_000 }),
            modalConfirm.click(),
          ]);
        } else {
          await page.waitForLoadState('networkidle');
        }

        // DB verify removed (hard delete) atau IsActive=0 (soft delete) — both acceptable
        const stillActive = await db.queryScalar(
          `SELECT COUNT(*) FROM AssessmentCategories WHERE Name = N'${escapeSql(editedName)}' AND (IsActive IS NULL OR IsActive = 1)`
        );
        expect(stillActive).toBe(0);
      });

      test('U4 — HC duplicate name rejected via TempData alert-danger', async ({ page }) => {
        test.setTimeout(FLOW_TIMEOUT_MS);
        // Pick existing category name dari DB (yang BUKAN dari U1-U3 — pakai seed `OJT` standard)
        const existingName = await db.queryString(
          `SELECT TOP 1 Name FROM AssessmentCategories WHERE (IsActive IS NULL OR IsActive = 1) ORDER BY Id ASC`
        );
        expect(existingName.length).toBeGreaterThan(0);

        await login(page, 'hc');
        await page.goto('/AssessmentAdmin/ManageCategories');
        await page.waitForLoadState('networkidle');

        // Open add form
        await page.locator('button[data-bs-target="#addCategoryForm"]').first().click();
        await expect(page.locator('#addCategoryForm.show')).toBeVisible({ timeout: 5_000 });

        // Fill DUPLICATE name
        await page.fill('#addCategoryForm input[name="name"]', existingName);
        const passInput = page.locator('#addCategoryForm input[name="defaultPassPercentage"]');
        if (await passInput.count() > 0) await passInput.fill('80');

        const submitBtn = page.locator('#addCategoryForm button[type="submit"]').filter({ hasText: /tambah|simpan|save/i }).first();
        await Promise.all([
          page.waitForURL(/\/ManageCategories/, { timeout: 10_000 }),
          submitBtn.click(),
        ]);

        // Pitfall 5: TempData-based reject — alert-danger, NOT inline ModelState validation

        const dupCount = await db.queryScalar(
          `SELECT COUNT(*) FROM AssessmentCategories WHERE Name = N'${escapeSql(existingName)}'`
        );
        expect(dupCount).toBe(1);
      });
    });
