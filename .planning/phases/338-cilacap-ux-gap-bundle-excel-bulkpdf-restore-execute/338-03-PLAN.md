    /// <summary>
    /// CIL-05: Sheet "Detail Per Soal" — grid per-peserta-per-soal.
    /// </summary>
    public static void AddDetailPerSoalSheet(
        XLWorkbook workbook,
        List<AssessmentSession> sessions,
        List<PackageUserResponse> responses,
        List<PackageQuestion> questions,
        IEnumerable<dynamic> sessionPackageMap)
    {
        var ws = workbook.Worksheets.Add("Detail Per Soal");
        // Header: No | Nama | NIP | Soal 1 Jawaban | Soal 1 Benar? | ... | Skor Total
        var sortedQuestions = questions.OrderBy(q => q.Id).ToList();

        // Header row
        ws.Cell(1, 1).Value = "No";
        ws.Cell(1, 2).Value = "Nama";
        ws.Cell(1, 3).Value = "NIP";
        int col = 4;
        foreach (var q in sortedQuestions)
        {
            ws.Cell(1, col++).Value = $"Soal {q.Id} Jawaban";
            ws.Cell(1, col++).Value = $"Soal {q.Id} Benar?";
        }
        ws.Cell(1, col).Value = "Skor Total";

        // Style header
        var headerRange = ws.Range(1, 1, 1, col);
        headerRange.Style.Font.Bold = true;
        headerRange.Style.Fill.BackgroundColor = XLColor.LightGray;

        // Data rows
        int rowIdx = 2;
        int seq = 1;
        foreach (var session in sessions)
        {
            ws.Cell(rowIdx, 1).Value = seq++;
            ws.Cell(rowIdx, 2).Value = session.User?.FullName ?? "";
            ws.Cell(rowIdx, 3).Value = session.User?.NIP ?? "";
            int c = 4;
            foreach (var q in sortedQuestions)
            {
                var response = responses.FirstOrDefault(r =>
                    r.AssessmentSessionId == session.Id && r.PackageQuestionId == q.Id);

                if (response == null)
                {
                    ws.Cell(rowIdx, c++).Value = "—";
                    ws.Cell(rowIdx, c++).Value = "—";
                }
                else
                {
                    var selectedOption = q.Options?.FirstOrDefault(o => o.Id == response.SelectedOptionId);
                    ws.Cell(rowIdx, c++).Value = selectedOption?.OptionText ?? response.AnswerText ?? "—";

                    var benar = response.IsCorrect == true ? "✓" : (response.IsCorrect == false ? "✗" : "—");
                    var cell = ws.Cell(rowIdx, c++);
                    cell.Value = benar;
                    if (response.IsCorrect == true) cell.Style.Font.FontColor = XLColor.Green;
                    else if (response.IsCorrect == false) cell.Style.Font.FontColor = XLColor.Red;
                }
            }
            ws.Cell(rowIdx, c).Value = session.Score ?? 0;
            rowIdx++;
        }

        ws.Columns().AdjustToContents();
    }

    /// <summary>
    /// CIL-05: Sheet "Elemen Teknis" — matrix peserta x elemen.
    /// </summary>
    public static void AddElemenTeknisSheet(
        XLWorkbook workbook,
        List<AssessmentSession> sessions,
        List<SessionElemenTeknisScore> etScores)
    {
        var ws = workbook.Worksheets.Add("Elemen Teknis");

        // Distinct elemen names urut alfabet
        var elements = etScores
            .Select(et => et.ElemenTeknis)
            .Where(e => !string.IsNullOrEmpty(e))
            .Distinct()
            .OrderBy(e => e)
            .ToList();

        // Header: Nama | NIP | Elemen 1 | Elemen 2 | ... | Avg
        ws.Cell(1, 1).Value = "Nama";
        ws.Cell(1, 2).Value = "NIP";
        int col = 3;
        foreach (var elem in elements)
            ws.Cell(1, col++).Value = elem;
        ws.Cell(1, col).Value = "Avg";

        var headerRange = ws.Range(1, 1, 1, col);
        headerRange.Style.Font.Bold = true;
        headerRange.Style.Fill.BackgroundColor = XLColor.LightGray;

        // Data rows
        int rowIdx = 2;
        foreach (var session in sessions)
        {
            ws.Cell(rowIdx, 1).Value = session.User?.FullName ?? "";
            ws.Cell(rowIdx, 2).Value = session.User?.NIP ?? "";
            int c = 3;
            var sessionScores = new List<decimal>();
            foreach (var elem in elements)
            {
                var score = etScores.FirstOrDefault(et =>
                    et.AssessmentSessionId == session.Id && et.ElemenTeknis == elem);
                if (score != null)
                {
                    ws.Cell(rowIdx, c++).Value = score.Score;
                    sessionScores.Add(score.Score);
                }
                else
                {
                    ws.Cell(rowIdx, c++).Value = "—";
                }
            }
            ws.Cell(rowIdx, c).Value = sessionScores.Any()
                ? Math.Round(sessionScores.Average(), 2)
                : 0;
            rowIdx++;
        }

        ws.Columns().AdjustToContents();
    }
}

    {
        var ws = workbook.Worksheets.Add("Detail Per Soal");
