    ```csharp
    namespace HcPortal.Helpers
    {
        public static class OrgTreePreOrder
        {
            /// <summary>
            /// Pre-order DFS flatten of an org-unit tree (ORG-TREE-01 / TEST-03).
            /// Mirrors wwwroot/js/orgTree.js buildTree + flattenTreePreOrder.
            /// Sibling order = DisplayOrder then Name (matches EF GetOrganizationTree ordering).
            /// Roots = ParentId == null, each emitted then immediately followed by ALL descendants.
            /// </summary>
            public static List<(int Id, int Depth)> BuildPreOrder(
                IReadOnlyList<(int Id, int? ParentId, int DisplayOrder, string Name)> flat)
            {
                var byParent = flat
                    .OrderBy(n => n.DisplayOrder).ThenBy(n => n.Name)
                    .GroupBy(n => n.ParentId)
                    .ToDictionary(g => g.Key, g => g.ToList());
                var outList = new List<(int, int)>();
                void Walk(int? parentId, int depth)
                {
                    if (!byParent.TryGetValue(parentId, out var kids)) return;
                    foreach (var k in kids) { outList.Add((k.Id, depth)); Walk(k.Id, depth + 1); }
                }
                Walk(null, 0);
                return outList;
            }
        }
    }
    ```
