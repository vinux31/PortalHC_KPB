**Change 3: Add GetEligibleCoachees endpoint**

Add this new action to AdminController (near other AJAX/JSON-returning actions, or before the nested DTO classes at the bottom of the file):

```csharp
/// <summary>
/// AJAX: Returns coachees eligible for a Proton exam — assigned to the track + 100% deliverables Approved.
/// Called from CreateAssessment form JS when category=Assessment Proton and track is selected.
/// </summary>
[HttpGet]
[Authorize(Roles = "Admin, HC")]
public async Task<IActionResult> GetEligibleCoachees(int protonTrackId)
{
    if (protonTrackId <= 0) return Json(new List<object>());

    // Coachees with active assignment to this track
    var assignedCoacheeIds = await _context.ProtonTrackAssignments
        .Where(a => a.ProtonTrackId == protonTrackId && a.IsActive)
        .Select(a => a.CoacheeId)
        .Distinct()
        .ToListAsync();

    if (!assignedCoacheeIds.Any()) return Json(new List<object>());

    // All deliverable IDs for this track (via Kompetensi → SubKompetensi → Deliverable)
    var trackDeliverableIds = await _context.ProtonKompetensiList
        .Where(k => k.ProtonTrackId == protonTrackId)
        .SelectMany(k => k.SubKompetensiList)
        .SelectMany(s => s.Deliverables)
        .Select(d => d.Id)
        .ToListAsync();

    if (!trackDeliverableIds.Any()) return Json(new List<object>());

    // Batch-load progress records for all assigned coachees on this track's deliverables
    var progressRecords = await _context.ProtonDeliverableProgresses
        .Where(p => assignedCoacheeIds.Contains(p.CoacheeId)
                 && trackDeliverableIds.Contains(p.ProtonDeliverableId))
        .Select(p => new { p.CoacheeId, p.ProtonDeliverableId, p.Status })
        .ToListAsync();

    // Eligible = has exactly trackDeliverableIds.Count Approved progress records
    var eligibleCoacheeIds = assignedCoacheeIds
        .Where(id =>
        {
            var mine = progressRecords.Where(p => p.CoacheeId == id).ToList();
            return mine.Count == trackDeliverableIds.Count && mine.All(p => p.Status == "Approved");
        })
        .ToList();

    if (!eligibleCoacheeIds.Any()) return Json(new List<object>());

    var users = await _context.Users
        .Where(u => eligibleCoacheeIds.Contains(u.Id))
        .Select(u => new { u.Id, u.FullName, u.Email, u.NIP, u.Section })
        .OrderBy(u => u.FullName)
        .ToListAsync();

    return Json(users);
}
```

```javascript
// ===== Assessment Proton adaptive form =====
(function() {
    var categorySelect = document.getElementById('Category');
    var protonSection = document.getElementById('protonFieldsSection');
    var protonTrackSelect = document.getElementById('protonTrackSelect');
    var protonEligibleSection = document.getElementById('protonEligibleSection');
    var protonEligibleStatus = document.getElementById('protonEligibleStatus');
    var protonCbContainer = document.getElementById('protonUserCheckboxContainer');
    var durationWrapper = document.getElementById('durationFieldWrapper');
    var passWrapper = document.getElementById('passPercentageWrapper');
    var standardUserCol = document.querySelector('#userCheckboxContainer')?.closest('.col-md-8');

    function applyProtonMode(isProton) {
        if (isProton) {
            if (protonSection) protonSection.classList.remove('d-none');
            if (standardUserCol) standardUserCol.style.display = 'none';
        } else {
            if (protonSection) protonSection.classList.add('d-none');
            if (standardUserCol) standardUserCol.style.display = '';
            // Restore duration + passPercentage
            if (durationWrapper) { durationWrapper.classList.remove('d-none'); durationWrapper.querySelector('input')?.setAttribute('required','required'); }
            if (passWrapper) { passWrapper.classList.remove('d-none'); passWrapper.querySelector('input')?.setAttribute('required','required'); }
            // Reset proton state
            if (protonTrackSelect) protonTrackSelect.value = '';
            if (protonEligibleSection) protonEligibleSection.classList.add('d-none');
            if (protonCbContainer) { protonCbContainer.innerHTML = ''; protonCbContainer.classList.add('d-none'); }
        }
    }

    if (categorySelect) {
        categorySelect.addEventListener('change', function() {
            applyProtonMode(this.value === 'Assessment Proton');
        });
        // On page load, apply if Assessment Proton already selected (e.g. form re-display after error)
        if (categorySelect.value === 'Assessment Proton') applyProtonMode(true);
    }

    if (protonTrackSelect) {
        protonTrackSelect.addEventListener('change', function() {
            var trackId = parseInt(this.value) || 0;
            var selectedOption = this.options[this.selectedIndex];
            var tahun = selectedOption ? (selectedOption.dataset.tahun || '') : '';

            // Tahun 3 = interview: hide Duration + PassPercentage
            var isTahun3 = tahun === 'Tahun 3';
            if (durationWrapper) {
                durationWrapper.classList.toggle('d-none', isTahun3);
                var durInput = durationWrapper.querySelector('input');
                if (durInput) { if (isTahun3) durInput.removeAttribute('required'); else durInput.setAttribute('required','required'); }
            }
            if (passWrapper) {
                passWrapper.classList.toggle('d-none', isTahun3);
            }

            if (!trackId) {
                if (protonEligibleSection) protonEligibleSection.classList.add('d-none');
                return;
            }

            // Load eligible coachees via AJAX
            if (protonEligibleSection) protonEligibleSection.classList.remove('d-none');
            if (protonEligibleStatus) { protonEligibleStatus.classList.remove('d-none'); protonEligibleStatus.innerHTML = '<i class="bi bi-hourglass-split me-1"></i>Memuat coachee eligible...'; }
            if (protonCbContainer) { protonCbContainer.innerHTML = ''; protonCbContainer.classList.add('d-none'); }

            fetch('/Admin/GetEligibleCoachees?protonTrackId=' + trackId)
                .then(function(r) { return r.json(); })
                .then(function(users) {
                    if (!protonCbContainer || !protonEligibleStatus) return;
                    if (!users || users.length === 0) {
                        protonEligibleStatus.innerHTML = '<i class="bi bi-info-circle me-1"></i>Tidak ada coachee eligible untuk track ini. Semua deliverable harus Approved.';
                        protonCbContainer.classList.add('d-none');
                    } else {
                        protonEligibleStatus.innerHTML = '<i class="bi bi-check-circle text-success me-1"></i>' + users.length + ' coachee eligible ditemukan.';
                        protonCbContainer.innerHTML = '';
                        users.forEach(function(u) {
                            var div = document.createElement('div');
                            div.className = 'form-check user-check-item mb-1';
                            div.dataset.name = u.fullName || '';
                            div.dataset.section = u.section || '';
                            div.innerHTML = '<input class="form-check-input user-checkbox" type="checkbox" name="UserIds" value="' + u.id + '" id="pu_' + u.id + '" />'
                                + '<label class="form-check-label" for="pu_' + u.id + '"><strong>' + (u.fullName || '-') + '</strong> <span class="text-muted">(' + (u.email || '') + ')</span></label>';
                            protonCbContainer.appendChild(div);
                        });
                        protonCbContainer.classList.remove('d-none');
                        protonCbContainer.addEventListener('change', function() {
                            var count = protonCbContainer.querySelectorAll('.user-checkbox:checked').length;
                            var badge = document.getElementById('selectedCountBadge');
                            if (badge) badge.textContent = count + ' selected';
                        });
                    }
                })
                .catch(function() {
                    if (protonEligibleStatus) protonEligibleStatus.innerHTML = '<i class="bi bi-exclamation-triangle text-danger me-1"></i>Gagal memuat daftar coachee.';
                });
        });
    }

    // Patch form submit: skip duration check for Tahun 3
    var form = document.getElementById('createAssessmentForm');
    if (form) {
        form.addEventListener('submit', function() {
            if (categorySelect && categorySelect.value === 'Assessment Proton') {
                var durWrapper = document.getElementById('durationFieldWrapper');
                if (durWrapper && durWrapper.classList.contains('d-none')) {
                    // Tahun 3: set DurationMinutes to 0 before submit

                    if (durInput) durInput.value = '0';
                }
            }
        }, true); // capture phase — runs before existing submit handler
    }
})();
```
