9. **JavaScript for AJAX coachee switching** (at bottom in @section Scripts):
```razor
@section Scripts {
<script>
document.addEventListener('DOMContentLoaded', function() {
    const coacheeSelect = document.getElementById('coacheeSelect');
    if (!coacheeSelect) return;

    coacheeSelect.addEventListener('change', async function() {
        const coacheeId = this.value;
        const loadingSpinner = document.getElementById('loadingSpinner');
        const tableSection = document.getElementById('tableSection');
        const statsSection = document.getElementById('statsSection');
        const coacheeInfo = document.getElementById('coacheeInfo');
        const messageArea = document.getElementById('messageArea');

        // Clear messages
        messageArea.innerHTML = '';

        if (!coacheeId) {
            tableSection.classList.add('d-none');
            statsSection.classList.add('d-none');
            coacheeInfo.classList.add('d-none');
            return;
        }

        // Show loading
        loadingSpinner.classList.remove('d-none');
        tableSection.classList.add('d-none');
        statsSection.classList.add('d-none');

        try {
            const response = await fetch(`@Url.Action("GetCoacheeDeliverables", "CDP")?coacheeId=${encodeURIComponent(coacheeId)}`);
            const result = await response.json();

            if (result.error) {
                messageArea.innerHTML = '<div class="alert alert-danger">Akses tidak diizinkan</div>';
                loadingSpinner.classList.add('d-none');
                return;
            }

            // Update coachee info
            document.getElementById('coacheeNameDisplay').textContent = result.coacheeName;
            const trackBadge = document.getElementById('trackLabelDisplay');
            if (result.trackLabel) {
                trackBadge.textContent = result.trackLabel;
                trackBadge.classList.remove('d-none');
            } else {
                trackBadge.classList.add('d-none');
            }
            coacheeInfo.classList.remove('d-none');

            // Handle error states
            if (result.noTrack) {
                messageArea.innerHTML = '<div class="alert alert-warning">Coachee ini belum memiliki penugasan track</div>';
                loadingSpinner.classList.add('d-none');
                return;
            }
            if (result.noProgress) {
                messageArea.innerHTML = '<div class="alert alert-info">Data progress tidak ditemukan</div>';
                loadingSpinner.classList.add('d-none');
                return;
            }

            // Update stats
            document.getElementById('statProgress').textContent = result.stats.progressPercent + '%';
            document.getElementById('progressBar').style.width = result.stats.progressPercent + '%';
            document.getElementById('statPendingActions').textContent = result.stats.pendingActions;
            document.getElementById('statPendingApprovals').textContent = result.stats.pendingApprovals;
            statsSection.classList.remove('d-none');

            // Build table rows with rowspan merging
            const tbody = document.getElementById('progressTableBody');
            tbody.innerHTML = '';

            // Group items by kompetensi then subKompetensi for rowspan
            const kompGroups = {};
            result.items.forEach(item => {
                if (!kompGroups[item.kompetensi]) kompGroups[item.kompetensi] = {};
                if (!kompGroups[item.kompetensi][item.subKompetensi])
                    kompGroups[item.kompetensi][item.subKompetensi] = [];
                kompGroups[item.kompetensi][item.subKompetensi].push(item);
            });

            for (const [komp, subGroups] of Object.entries(kompGroups)) {
                let kompRowCount = 0;
                for (const items of Object.values(subGroups)) kompRowCount += items.length;
                let firstKomp = true;

                for (const [sub, items] of Object.entries(subGroups)) {
                    let firstSub = true;
                    items.forEach(item => {
                        const tr = document.createElement('tr');
                        let cells = '';

                        if (firstKomp) {
                            cells += `<td rowspan="${kompRowCount}" class="align-middle fw-bold">${escapeHtml(komp)}</td>`;
                            firstKomp = false;
                        }
                        if (firstSub) {
                            cells += `<td rowspan="${items.length}" class="align-middle">${escapeHtml(sub)}</td>`;
                            firstSub = false;
                        }

                        cells += `<td>${escapeHtml(item.deliverable)}</td>`;
                        cells += `<td class="text-center">${item.evidenceStatus === 'Uploaded'
                            ? '<span class="badge bg-success">Uploaded</span>'
                            : '<span class="badge bg-secondary">Pending</span>'}</td>`;
                        cells += `<td class="text-center">${getApprovalBadge(item.approvalSrSpv)}</td>`;
                        cells += `<td class="text-center">${getApprovalBadge(item.approvalSectionHead)}</td>`;
                        cells += `<td class="text-center">${getApprovalBadge(item.approvalHC)}</td>`;
                        cells += `<td class="text-center">
                            <div class="dropdown">
                                <button class="btn btn-sm btn-outline-secondary dropdown-toggle" type="button" data-bs-toggle="dropdown">
                                    <i class="bi bi-three-dots-vertical"></i>
                                </button>
                                <ul class="dropdown-menu">
                                    <li><a class="dropdown-item" href="/CDP/Deliverable/${item.id}"><i class="bi bi-eye me-1"></i>Lihat Detail</a></li>
                                </ul>
                            </div>
                        </td>`;

                        tr.innerHTML = cells;
                        tbody.appendChild(tr);
                    });
                }
            }

            tableSection.classList.remove('d-none');
        } catch (err) {
            messageArea.innerHTML = '<div class="alert alert-danger">Terjadi kesalahan saat memuat data. Silakan coba lagi.</div>';
        } finally {
            loadingSpinner.classList.add('d-none');
        }
    });

    function getApprovalBadge(status) {
        switch (status) {
            case 'Approved': return '<span class="badge bg-success">Approved</span>';
            case 'Rejected': return '<span class="badge bg-danger">Rejected</span>';
            case 'Pending': return '<span class="badge bg-warning text-dark">Pending</span>';
            default: return '<span class="badge bg-secondary">Not Started</span>';
        }
    }

    function escapeHtml(text) {
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
});
</script>
}
```
