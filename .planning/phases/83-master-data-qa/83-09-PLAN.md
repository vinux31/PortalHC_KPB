**3. Add JS event handlers** for the new buttons. Add after the existing `btn-delete-row` click handler:

```javascript
// Nonaktifkan silabus row
$(document).on('click', '.btn-deactivate-row', async function () {
    const id = $(this).data('id');
    const name = $(this).data('name');
    if (!confirm(`Nonaktifkan silabus "${name}"?\n\nSilabus tidak akan muncul di Plan IDP dan Coaching Proton. Dapat diaktifkan kembali.`)) return;
    const token = $('input[name="__RequestVerificationToken"]').val();
    const resp = await fetch('/ProtonData/SilabusDeactivate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'RequestVerificationToken': token },
        body: JSON.stringify({ KompetensiId: id })
    });
    const result = await resp.json();
    if (result.success) {
        showToast(result.message, 'success');
        setTimeout(() => window.location.reload(), 800);
    } else {
        showToast(result.message || 'Gagal menonaktifkan silabus.', 'danger');
    }
});

// Aktifkan kembali silabus row
$(document).on('click', '.btn-reactivate-row', async function () {
    const id = $(this).data('id');
    const name = $(this).data('name');
    const token = $('input[name="__RequestVerificationToken"]').val();
    const resp = await fetch('/ProtonData/SilabusReactivate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'RequestVerificationToken': token },
        body: JSON.stringify({ KompetensiId: id })
    });
    const result = await resp.json();
    if (result.success) {
        showToast(result.message, 'success');
        setTimeout(() => window.location.reload(), 800);
    } else {
        showToast(result.message || 'Gagal mengaktifkan silabus.', 'danger');
    }
});
```
