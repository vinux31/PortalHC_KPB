function escHtml(s) {
    return String(s || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function showToast(type, msg) {
    if (type === 'success') {
        var el = document.getElementById('saveToast');
        if (msg) el.querySelector('.toast-body').innerHTML = '<i class="bi bi-check-circle-fill me-2"></i>' + escHtml(msg);
        bootstrap.Toast.getOrCreateInstance(el, { delay: 2000 }).show();
    } else {
        var eel = document.getElementById('errToast');
        document.getElementById('errToastMsg').textContent = msg || 'Terjadi error.';
        bootstrap.Toast.getOrCreateInstance(eel, { delay: 3500 }).show();
    }
}

function getToken() {
    return document.querySelector('input[name="__RequestVerificationToken"]').value;
}

function renderReadTable(bagianName) {
    var panel = document.getElementById('readTablePanel');
    if (!panel) return;

    var bagian = kkjBagians.find(function(b) { return b.Name === bagianName; });
    var columns = bagian ? (bagian.Columns || []) : [];
    var items = kkjItems.filter(function(i) { return i.Bagian === bagianName; })
                        .slice().sort(function(a, b) { return (a.No || 0) - (b.No || 0); });

    // Build thead
    var theadCells = '<th style="width:45px;">No</th>'
        + '<th style="width:80px;">Indeks</th>'
        + '<th style="min-width:180px;">Kompetensi</th>'
        + '<th style="min-width:110px;">SkillGroup</th>'
        + '<th style="min-width:110px;">SubSkillGroup</th>';
    columns.forEach(function(col) {
        theadCells += '<th class="col-target text-center">' + escHtml(col.Name) + '</th>';
    });
    if (columns.length === 0) {
        theadCells += '<th class="text-muted">Belum ada kolom</th>';
    }
    theadCells += '<th style="width:65px;">Aksi</th>';

    // Build tbody
    var tbodyRows = '';
    if (items.length === 0) {
        var span = 5 + columns.length + 1;
        tbodyRows = '<tr><td colspan="' + span + '" class="text-center text-muted py-4">Belum ada item di bagian ini.</td></tr>';
    } else {
        items.forEach(function(item) {
            var tds = '<td class="text-center">' + escHtml(item.No) + '</td>'
                + '<td>' + escHtml(item.Indeks) + '</td>'
                + '<td>' + escHtml(item.Kompetensi) + '</td>'
                + '<td class="text-muted">' + escHtml(item.SkillGroup) + '</td>'
                + '<td class="text-muted">' + escHtml(item.SubSkillGroup) + '</td>';
            columns.forEach(function(col) {
                var tv = item.TargetValues && item.TargetValues.find(function(t) { return t.KkjColumnId === col.Id; });
                var val = tv ? (tv.Value || '-') : '-';
                tds += '<td class="col-target text-center">' + escHtml(val) + '</td>';
            });
            if (columns.length === 0) {
                tds += '<td class="text-muted text-center">-</td>';
            }
            var safeK = escHtml(item.Kompetensi).replace(/'/g, "\\'");
            tds += '<td class="text-center">'
                + '<button class="btn btn-outline-danger btn-sm" onclick="deleteRow(' + item.Id + ', \'' + safeK + '\')">'
                + '<i class="bi bi-trash"></i></button>'
                + '</td>';
            tbodyRows += '<tr data-id="' + item.Id + '">' + tds + '</tr>';
        });
    }

    panel.innerHTML = '<div class="table-responsive custom-scrollbar mb-3">'
        + '<table class="table table-sm table-bordered table-hover mb-0">'
        + '<thead class="table-dark"><tr>' + theadCells + '</tr></thead>'
        + '<tbody>' + tbodyRows + '</tbody>'
        + '</table></div>';
}

document.getElementById('btnRenameBagian') && document.getElementById('btnRenameBagian').addEventListener('click', function() {
    withFilter(function(filter, currentName) {
        if (!currentName) return;
        var newName = prompt('Nama baru untuk bagian "' + currentName + '":', currentName);
        if (!newName || newName.trim() === '' || newName.trim() === currentName) return;
        newName = newName.trim();
        var bagian = kkjBagians.find(function(b) { return b.Name === currentName; });
        if (!bagian) return;
        var payload = [{ Id: bagian.Id, Name: newName, DisplayOrder: bagian.DisplayOrder }];
        $.ajax({
            url: '/Admin/KkjBagianSave', type: 'POST', contentType: 'application/json',
            headers: { 'RequestVerificationToken': getToken() },
            data: JSON.stringify(payload),
            success: function(resp) {
                if (resp.success) {
                    bagian.Name = newName;
                    kkjItems.forEach(function(i) { if (i.Bagian === currentName) i.Bagian = newName; });
                    var opt = filter.querySelector('option[value="' + currentName + '"]');
                    if (opt) { opt.value = newName; opt.textContent = newName; }
                    filter.value = newName;
                    renderReadTable(newName);
                    var link = document.getElementById('linkViewCmp');
                    if (link) link.href = '/CMP/Kkj?section=' + encodeURIComponent(newName);
                } else { showToast('error', resp.message); }
            },
            error: function(xhr) { showToast('error', 'Server error: ' + xhr.statusText); }
        });
    });
});

document.getElementById('btnDeleteBagian') && document.getElementById('btnDeleteBagian').addEventListener('click', function() {
    withFilter(function(filter, currentName) {
        if (!currentName) return;
        var bagian = kkjBagians.find(function(b) { return b.Name === currentName; });
        if (!bagian) return;
        if (!confirm('Hapus bagian "' + currentName + '"? Tidak dapat dihapus jika masih ada item.')) return;
        $.ajax({
            url: '/Admin/KkjBagianDelete', type: 'POST',
            data: { id: bagian.Id, __RequestVerificationToken: getToken() },
            success: function(resp) {
                if (resp.success) {
                    kkjBagians = kkjBagians.filter(function(b) { return b.Id !== bagian.Id; });
                    var opt = filter.querySelector('option[value="' + currentName + '"]');
                    if (opt) opt.remove();
                    if (kkjBagians.length > 0) {
                        filter.value = kkjBagians[0].Name;
                        renderReadTable(kkjBagians[0].Name);
                    } else {
                        document.getElementById('readTablePanel').innerHTML = '<p class="text-muted">Tidak ada bagian.</p>';
                    }
                } else { showToast('error', resp.message); }
            },
            error: function(xhr) { showToast('error', 'Server error: ' + xhr.statusText); }
        });
    });
});

document.getElementById('btnAddBagianRead') && document.getElementById('btnAddBagianRead').addEventListener('click', function() {
    $.ajax({
        url: '/Admin/KkjBagianAdd', type: 'POST',
        data: { __RequestVerificationToken: getToken() },
        success: function(resp) {
            if (resp.success) {

                renderReadTable(resp.name);
                loadColumnPanel(resp.id);
                loadMappingPanel(resp.id);
            } else { showToast('error', resp.message); }
        },
        error: function(xhr) { showToast('error', 'Server error: ' + xhr.statusText); }
    });
});

function deleteRow(id, kompetensi) {
    if (!confirm('Hapus "' + kompetensi + '"?')) return;
    $.ajax({
        url: '/Admin/KkjMatrixDelete', type: 'POST',
        data: { id: id, __RequestVerificationToken: getToken() },
        success: function(resp) {
            if (resp.success) {
                kkjItems = kkjItems.filter(function(i) { return i.Id !== id; });
                var filter = document.getElementById('bagianFilter');
                if (filter) renderReadTable(filter.value);
            } else if (resp.blocked) {
                showToast('error', resp.message);
            } else { showToast('error', resp.message); }
        }
    });
}

// makeRow: create a <tr> with edit inputs
// ======================================================
function makeRow(item, columns) {
    var tr = document.createElement('tr');
    tr.dataset.id = item.Id || 0;

    // Hidden bagian field (not a <td>)
    var bagianHidden = document.createElement('input');
    bagianHidden.type = 'hidden';
    bagianHidden.name = 'Bagian';
    bagianHidden.value = item.Bagian || '';
    tr.appendChild(bagianHidden);

    // Fixed editable fields
    var fixed = [
        { name: 'No',            val: item.No,            cls: 'col-no',   type: 'number' },
        { name: 'SkillGroup',    val: item.SkillGroup,    cls: 'col-meta' },
        { name: 'SubSkillGroup', val: item.SubSkillGroup, cls: 'col-meta' },
        { name: 'Indeks',        val: item.Indeks,        cls: 'col-meta' },
        { name: 'Kompetensi',    val: item.Kompetensi,    cls: 'col-meta', style: 'min-width:180px' }
    ];
    fixed.forEach(function(f) {
        var td = document.createElement('td');
        td.className = f.cls || '';
        if (f.style) td.style.cssText = f.style;
        var input = document.createElement('input');
        input.type = f.type || 'text';
        input.name = f.name;
        input.value = (f.val === null || f.val === undefined) ? '' : f.val;
        input.className = 'edit-input';
        td.appendChild(input);
        tr.appendChild(td);
    });

    // Dynamic target value columns — with validation
    (columns || []).forEach(function(col) {
        var td = document.createElement('td');
        td.className = 'col-target';
        var tv = item.TargetValues && item.TargetValues.find(function(t) { return t.KkjColumnId === col.Id; });
        var input = document.createElement('input');
        input.type = 'text';
        input.className = 'edit-input text-center';
        input.dataset.columnId = col.Id;
        input.value = tv ? (tv.Value || '-') : '-';
        input.maxLength = 1;
        // Validation: only 1-5 or -
        input.addEventListener('change', function() {
            var v = input.value.trim();
            var valid = (v === '-' || v === '1' || v === '2' || v === '3' || v === '4' || v === '5');
            if (!valid) {
                input.classList.add('is-invalid');
                input.value = '-';
                setTimeout(function() { input.classList.remove('is-invalid'); }, 1500);
            }
        });
        input.addEventListener('blur', function() {
            var v = input.value.trim();
            if (v === '') input.value = '-';
        });
        td.appendChild(input);
        tr.appendChild(td);
    });

    // Aksi column: insert-below + delete
    var tdAksi = document.createElement('td');
    tdAksi.className = 'col-aksi';

    var btnInsert = document.createElement('button');
    btnInsert.type = 'button';
    btnInsert.className = 'btn btn-outline-success btn-sm me-1';
    btnInsert.title = 'Sisipkan baris baru di bawah';
    btnInsert.innerHTML = '<i class="bi bi-plus-circle"></i>';
    btnInsert.addEventListener('click', function() {
        var bagianName = tr.querySelector('input[name="Bagian"]').value || '';
        var rowBagian = kkjBagians.find(function(b) { return b.Name === bagianName; });
        var rowCols = rowBagian ? (rowBagian.Columns || []) : (columns || []);
        var newRow = makeEmptyRow(rowCols);
        newRow.querySelector('input[name="Bagian"]').value = bagianName;
        tr.parentNode.insertBefore(newRow, tr.nextSibling);
    });

    var btnDel = document.createElement('button');
    btnDel.type = 'button';
    btnDel.className = 'btn btn-outline-danger btn-sm';
    btnDel.title = 'Hapus baris ini';
    btnDel.innerHTML = '<i class="bi bi-trash"></i>';
    btnDel.addEventListener('click', function() {
        var id = parseInt(tr.dataset.id) || 0;
        if (id === 0) {
            tr.remove();
        } else {
            var komp = tr.querySelector('[name="Kompetensi"]') ? tr.querySelector('[name="Kompetensi"]').value : '';
            if (!confirm('Hapus "' + komp + '"?')) return;
            $.ajax({
                url: '/Admin/KkjMatrixDelete', type: 'POST',
                data: { id: id, __RequestVerificationToken: getToken() },
                success: function(resp) {
                    if (resp.success) {
                        tr.remove();
                        kkjItems = kkjItems.filter(function(i) { return i.Id !== id; });
                    } else if (resp.blocked) {
                        showToast('error', resp.message);
                    } else {
                        showToast('error', resp.message);
                    }
                }
            });
        }
    });

    tdAksi.appendChild(btnInsert);
    tdAksi.appendChild(btnDel);
    tr.appendChild(tdAksi);
    return tr;
}

function makeEmptyRow(columns) {

    return makeRow({

        TargetValues: emptyTVs
    }, columns || []);
}

function collectBagians() {
    var result = [];
    document.querySelectorAll('.kkj-edit-tbl').forEach(function(tbl) {
        var bagianId = parseInt(tbl.dataset.bagianId) || 0;
        var b = { Id: bagianId, Name: '', DisplayOrder: 0 };
        var nameEl = document.querySelector('.bagian-name-input[data-bagian-id="' + bagianId + '"]');
        if (nameEl) {
            b.Name = nameEl.value;
            b.DisplayOrder = parseInt(nameEl.dataset.bagianOrder) || 0;
        }
        result.push(b);
    });
    return result;
}

function collectColumnRenames() {
    var renames = [];
    document.querySelectorAll('.col-name-input').forEach(function(inp) {
        renames.push({ Id: parseInt(inp.dataset.columnId), Name: inp.value.trim() });
    });
    return renames;
}

function collectRows() {
    var rows = [];
    document.querySelectorAll('.bagian-tbody tr').forEach(function(tr) {
        var getVal = function(name) {
            var el = tr.querySelector('input[name="' + name + '"]');
            return el ? el.value : '';
        };
        var targetValues = [];
        tr.querySelectorAll('input[data-column-id]').forEach(function(inp) {
            targetValues.push({ KkjColumnId: parseInt(inp.dataset.columnId), Value: inp.value || '-' });
        });
        rows.push({
            Id:            parseInt(tr.dataset.id) || 0,
            Bagian:        getVal('Bagian'),
            No:            parseInt(getVal('No')) || 0,
            SkillGroup:    getVal('SkillGroup'),
            SubSkillGroup: getVal('SubSkillGroup'),
            Indeks:        getVal('Indeks'),
            Kompetensi:    getVal('Kompetensi'),
            TargetValues:  targetValues
        });
    });
    return rows;
}

document.getElementById('btnSave').addEventListener('click', function() {
    var token = getToken();
    var bagians = collectBagians();
    var rows = collectRows();
    var renames = collectColumnRenames();

    document.getElementById('btnSave').disabled = true;

    // Step 0: save column renames
    var renamePromises = renames.map(function(r) {
        return $.ajax({
            url: '/Admin/KkjColumnSave', type: 'POST', contentType: 'application/json',
            headers: { 'RequestVerificationToken': token },
            data: JSON.stringify(r)
        });
    });

    var renameDeferred = renamePromises.length > 0
        ? $.when.apply($, renamePromises)
        : $.Deferred().resolve();

    renameDeferred.then(function() {
        // Step 1: save bagian names/order
        $.ajax({
            url: '/Admin/KkjBagianSave', type: 'POST', contentType: 'application/json',
            headers: { 'RequestVerificationToken': token },
            data: JSON.stringify(bagians),
            success: function(bagianResp) {
                if (!bagianResp.success) {
                    document.getElementById('btnSave').disabled = false;
                    showToast('error', 'Error menyimpan bagian: ' + bagianResp.message);
                    return;
                }
                if (rows.length === 0) {
                    document.getElementById('btnSave').disabled = false;
                    showToast('success', 'Data berhasil disimpan.');
                    setTimeout(function() { location.reload(); }, 1800);
                    return;
                }
                // Step 2: save rows
                $.ajax({
                    url: '/Admin/KkjMatrixSave', type: 'POST', contentType: 'application/json',
                    headers: { 'RequestVerificationToken': token },
                    data: JSON.stringify(rows),
                    success: function(rowResp) {
                        document.getElementById('btnSave').disabled = false;
                        if (rowResp.success) {
                            showToast('success', rowResp.message);
                            setTimeout(function() { location.reload(); }, 1800);
                        } else {
                            showToast('error', 'Error menyimpan baris: ' + rowResp.message);
                        }
                    },
                    error: function(xhr) {
                        document.getElementById('btnSave').disabled = false;
                        showToast('error', 'Server error (rows): ' + xhr.statusText);
                    }
                });
            },
            error: function(xhr) {
                document.getElementById('btnSave').disabled = false;
                showToast('error', 'Server error (bagian): ' + xhr.statusText);
            }
        });
    });
});

                    + '<td>'
                    + '<button class="btn btn-sm btn-outline-primary me-1 btn-save-col" title="Simpan"><i class="bi bi-check"></i></button>'

            html += '</tbody></table>';

                $.ajax({
                    url: '/Admin/KkjColumnSave', type: 'POST', contentType: 'application/json',
                    headers: { 'RequestVerificationToken': token },
                    data: JSON.stringify({ Id: colId, Name: name, DisplayOrder: order }),
                    success: function(resp) {
                        if (resp.success) {
                            // Update local cache
                            var b = kkjBagians.find(function(x) { return x.Id === bagianId; });
                            if (b) {
                                var c = (b.Columns || []).find(function(x) { return x.Id === colId; });
                                if (c) { c.Name = name; c.DisplayOrder = order; }
                            }
                            loadColumnPanel(bagianId);
                            var filter = document.getElementById('bagianFilter');
