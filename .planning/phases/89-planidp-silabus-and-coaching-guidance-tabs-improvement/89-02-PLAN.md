const orgStructure = @Html.Raw(ViewBag.OrgStructureJson ?? "{}");

**2. Silabus read-only table renderer** (adapted from ProtonData/Index.cshtml renderViewTable — remove Edit button and action column):
```javascript
(function() {
    const silabusRows = JSON.parse(document.getElementById('silabusData').textContent || '[]');
    const filterRaw = JSON.parse(document.getElementById('silabusFilter').textContent);
    const hasFilter = filterRaw.hasFilter;
    const currentUnit = filterRaw.unit || '';
    const container = document.getElementById('silabusTableContainer');

    function escHtml(str) {
        if (!str) return '';
        return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
    }

    if (!hasFilter) {
        container.innerHTML = `<div class="text-center py-5 text-muted">
            <i class="bi bi-funnel fs-1 mb-3 d-block opacity-50"></i>
            <p>Pilih Bagian, Unit, dan Track kemudian klik <strong>Muat Data</strong> untuk menampilkan silabus.</p>
        </div>`;
        return;
    }

    if (silabusRows.length === 0) {
        container.innerHTML = '<div class="alert alert-info">Tidak ada data silabus untuk filter ini.</div>';
        return;
    }

    let html = `<div class="mb-2 text-muted small">Menampilkan silabus untuk: <strong>${escHtml(currentUnit)}</strong></div>`;
    html += `<div class="table-responsive"><table class="table table-bordered table-sm">
        <thead class="table-light">
            <tr>
                <th style="width:60px">No</th>
                <th>Kompetensi</th>
                <th>Sub Kompetensi</th>
                <th>Deliverable</th>
            </tr>
        </thead>
        <tbody>`;

    let i = 0;
    while (i < silabusRows.length) {
        const row = silabusRows[i];
        // Kompetensi rowspan
        let kompSpan = 1;
        while (i + kompSpan < silabusRows.length && silabusRows[i + kompSpan].Kompetensi === row.Kompetensi) kompSpan++;

        let j = i;
        while (j < i + kompSpan) {
            const subRow = silabusRows[j];
            // SubKompetensi rowspan within kompetensi group
            let subSpan = 1;
            while (j + subSpan < i + kompSpan && silabusRows[j + subSpan].SubKompetensi === subRow.SubKompetensi) subSpan++;

            for (let k = j; k < j + subSpan; k++) {
                const dRow = silabusRows[k];
                html += '<tr>';
                if (k === i) {
                    html += `<td rowspan="${kompSpan}" class="align-middle fw-semibold text-center bg-light">${escHtml(dRow.No)}</td>`;
                    html += `<td rowspan="${kompSpan}" class="align-middle fw-semibold">${escHtml(dRow.Kompetensi)}</td>`;
                }
                if (k === j) {
                    html += `<td rowspan="${subSpan}" class="align-middle">${escHtml(dRow.SubKompetensi)}</td>`;
                }
                html += `<td>${escHtml(dRow.Deliverable)}</td>`;
                html += '</tr>';
            }
            j += subSpan;
        }
        i += kompSpan;
    }

    html += '</tbody></table></div>';
    container.innerHTML = html;
})();
```

**3. Coaching Guidance accordion renderer:**
```javascript
(function() {
    const guidanceData = JSON.parse(document.getElementById('guidanceData').textContent || '[]');
    const container = document.getElementById('guidanceAccordionContainer');

    function escHtml(str) {
        if (!str) return '';
        return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
    }
    function formatSize(bytes) {
        if (!bytes) return '';
        if (bytes < 1024) return bytes + ' B';
        if (bytes < 1024*1024) return (bytes/1024).toFixed(1) + ' KB';
        return (bytes/(1024*1024)).toFixed(1) + ' MB';
    }

    if (!guidanceData || guidanceData.length === 0) {
        container.innerHTML = '<div class="alert alert-info">Belum ada file coaching guidance.</div>';
        return;
    }

    // 4-level accordion: Bagian > Unit > TrackType > TahunKe
    let html = '<div class="accordion" id="guidanceAccordion">';
    guidanceData.forEach(function(bagianGroup, bi) {
        const bagianId = 'bagian_' + bi;
        html += `<div class="accordion-item">
            <h2 class="accordion-header">
                <button class="accordion-button collapsed fw-bold" type="button"
                        data-bs-toggle="collapse" data-bs-target="#${bagianId}">
                    <i class="bi bi-diagram-3 me-2 text-primary"></i>${escHtml(bagianGroup.Bagian)}
                </button>
            </h2>
            <div id="${bagianId}" class="accordion-collapse collapse" data-bs-parent="#guidanceAccordion">
                <div class="accordion-body p-0">
                    <div class="accordion" id="units_${bi}">`;

        bagianGroup.Units.forEach(function(unitGroup, ui) {
            const unitId = 'unit_' + bi + '_' + ui;
            html += `<div class="accordion-item border-0 border-bottom">
                <h2 class="accordion-header">
                    <button class="accordion-button collapsed ps-4" type="button"
                            data-bs-toggle="collapse" data-bs-target="#${unitId}">
                        <i class="bi bi-building me-2 text-secondary"></i>${escHtml(unitGroup.Unit)}
                    </button>
                </h2>
                <div id="${unitId}" class="accordion-collapse collapse" data-bs-parent="#units_${bi}">
                    <div class="accordion-body p-0">
                        <div class="accordion" id="types_${bi}_${ui}">`;

            unitGroup.TrackTypes.forEach(function(ttGroup, ti) {
                const ttId = 'tt_' + bi + '_' + ui + '_' + ti;
                html += `<div class="accordion-item border-0 border-bottom">
                    <h2 class="accordion-header">
                        <button class="accordion-button collapsed ps-5" type="button"
                                data-bs-toggle="collapse" data-bs-target="#${ttId}">
                            <i class="bi bi-person-badge me-2 text-info"></i>${escHtml(ttGroup.TrackType)}
                        </button>
                    </h2>
                    <div id="${ttId}" class="accordion-collapse collapse" data-bs-parent="#types_${bi}_${ui}">
                        <div class="accordion-body p-0">
                            <div class="accordion" id="tahun_${bi}_${ui}_${ti}">`;

                ttGroup.TahunKeList.forEach(function(tkGroup, ki) {
                    const tkId = 'tk_' + bi + '_' + ui + '_' + ti + '_' + ki;
                    html += `<div class="accordion-item border-0 border-bottom">
                        <h2 class="accordion-header">
                            <button class="accordion-button collapsed ps-5 text-dark" type="button"
                                    data-bs-toggle="collapse" data-bs-target="#${tkId}">
                                <i class="bi bi-calendar3 me-2 text-muted"></i>${escHtml(tkGroup.TahunKe)}
                                <span class="badge bg-secondary ms-2">${tkGroup.Files.length} file</span>
                            </button>
                        </h2>
                        <div id="${tkId}" class="accordion-collapse collapse" data-bs-parent="#tahun_${bi}_${ui}_${ti}">
                            <div class="accordion-body ps-5">`;

                    if (tkGroup.Files.length === 0) {
                        html += '<p class="text-muted small">Belum ada file coaching guidance.</p>';
                    } else {
                        html += '<div class="list-group list-group-flush">';
                        tkGroup.Files.forEach(function(f) {
                            html += `<div class="list-group-item d-flex align-items-center justify-content-between px-0">
                                <div>
                                    <i class="bi bi-file-earmark me-2 text-primary"></i>
                                    <span>${escHtml(f.FileName)}</span>
                                    <small class="text-muted ms-2">${formatSize(f.FileSize)} &bull; ${escHtml(f.UploadedAt)}</small>
                                </div>
                                <a href="/CDP/GuidanceDownload?id=${f.Id}" class="btn btn-sm btn-outline-primary" title="Download">
                                    <i class="bi bi-download me-1"></i>Download
                                </a>
                            </div>`;
                        });
                        html += '</div>';
                    }

                    html += `</div></div></div>`; // accordion-body, collapse, accordion-item (tahun)
                });

                html += `</div></div></div></div>`; // accordion (tahun), accordion-body, collapse, accordion-item (tracktype)
            });

            html += `</div></div></div></div>`; // accordion (types), accordion-body, collapse, accordion-item (unit)
        });

        html += `</div></div></div></div>`; // accordion (units), accordion-body, collapse, accordion-item (bagian)
    });

    html += '</div>'; // outer accordion
    container.innerHTML = html;
})();
```
