```csharp
public static async Task SeedCDPTestData(ApplicationDbContext context)
{
    // Check if already seeded
    if (context.ProtonTrackAssignments.Any())
        return;

    // Create users
    var coachee1 = new ApplicationUser { Id = "coachee1", UserName = "coachee1", RoleLevel = 1, Bagian = "Alkylation", Unit = "Operations", IsActive = true };
    var coach1 = new ApplicationUser { Id = "coach1", UserName = "coach1", RoleLevel = 3, Bagian = "Alkyation", Unit = "Operations", IsActive = true };
    var spv1 = new ApplicationUser { Id = "spv1", UserName = "spv1", RoleLevel = 4, Bagian = "Alkyation", Unit = "Operations", IsActive = true };

    // Create assignments
    context.ProtonTrackAssignments.AddRange(
        new ProtonTrackAssignment { CoacheeId = "coachee1", CoachId = "coach1", ProtonTrackId = 1, IsActive = true }
    );

    // Create deliverable progress with various statuses
    context.ProtonDeliverableProgresses.AddRange(
        new ProtonDeliverableProgress { CoacheeId = "coachee1", DeliverableId = 1, Status = "Pending" },
        new ProtonDeliverableProgress { CoacheeId = "coachee1", DeliverableId = 2, Status = "Completed" }
    );

    await context.SaveChangesAsync();
}
```
