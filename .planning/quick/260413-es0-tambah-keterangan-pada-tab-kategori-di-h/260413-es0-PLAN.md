```javascript
// Deskripsi kategori assessment
(function () {
    var categoryDescriptions = {
        'OJT': 'On the Job Training \u2014 pelatihan langsung di tempat kerja untuk meningkatkan keterampilan operasional.',
        'IHT': 'In House Training \u2014 pelatihan internal yang diselenggarakan di lingkungan perusahaan.',
        'Training Licencor': 'Pelatihan yang diselenggarakan oleh pemberi lisensi (licensor) untuk memenuhi standar teknologi berlisensi.',
        'OTS': 'Operator Training Simulator \u2014 pelatihan menggunakan simulator untuk operator kilang.',
        'Assessment Proton': 'Assessment terkait program Proton (Program Transformasi Operasional & Bisnis).',
        'Mandatory HSSE Training': 'Pelatihan wajib Health, Safety, Security & Environment yang harus dipenuhi seluruh pekerja.',
        'Gas Tester': 'Sertifikasi kemampuan pengujian gas untuk keselamatan area kerja kilang.'
    };

    function updateCategoryDescription(value) {
        var descDiv = document.getElementById('category-description');
        var descText = document.getElementById('category-description-text');
        if (!descDiv || !descText) return;
        if (value && categoryDescriptions[value]) {
            descText.textContent = categoryDescriptions[value];
            descDiv.style.display = '';
        } else {
            descDiv.style.display = 'none';
            descText.textContent = '';
        }
    }

    var categorySelect = document.getElementById('Category');
    if (categorySelect) {
        categorySelect.addEventListener('change', function () {
            updateCategoryDescription(this.value);
        });
        // Trigger on page load jika nilai sudah terisi (mis. validation re-render)
        updateCategoryDescription(categorySelect.value);
    }
})();
```
