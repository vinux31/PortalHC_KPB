6.3 Script SQL Cleanup (Reference untuk Tim IT)

Script berikut menghapus semua assessment dengan judul ber-prefix [TEST] beserta data terkait. Wajib backup DB dulu sebelum run.

-- ===================================================== -- CLEANUP DATA TEST ASSESSMENT — PortalHC KPB (Dev) -- Database: HcPortalDB_Dev -- Filter : AssessmentSession.Title LIKE '[TEST]%' -- Tanggal : diisi sebelum eksekusi -- ===================================================== USE HcPortalDB_Dev; GO -- 1. Ambil daftar AssessmentSessionId yang akan dihapus DECLARE @TestSessionIds TABLE (Id INT); INSERT INTO @TestSessionIds (Id) SELECT Id FROM AssessmentSessions WHERE Title LIKE '[TEST]%'; -- Preview (jalankan dulu untuk verify): SELECT * FROM AssessmentSessions WHERE Id IN (SELECT Id FROM @TestSessionIds); BEGIN TRANSACTION; -- 2. Hapus jawaban worker DELETE FROM PackageUserResponses WHERE AssessmentSessionId IN (SELECT Id FROM @TestSessionIds); -- 3. Hapus assignment worker-paket DELETE FROM UserPackageAssignments WHERE AssessmentSessionId IN (SELECT Id FROM @TestSessionIds); -- 4. Hapus log aktivitas exam (jika tabel ada) DELETE FROM ExamActivityLogs WHERE SessionId IN (SELECT Id FROM @TestSessionIds); -- 5. Hapus skor per elemen teknis DELETE FROM SessionElemenTeknisScores WHERE AssessmentSessionId IN (SELECT Id FROM @TestSessionIds); -- 6. Hapus attempt history DELETE FROM AssessmentAttemptHistory WHERE SessionId IN (SELECT Id FROM @TestSessionIds); -- 7. Hapus opsi soal (via Question → Package → Session) DELETE po FROM PackageOptions po INNER JOIN PackageQuestions pq ON po.PackageQuestionId = pq.Id INNER JOIN AssessmentPackages ap ON pq.AssessmentPackageId = ap.Id WHERE ap.AssessmentSessionId IN (SELECT Id FROM @TestSessionIds); -- 8. Hapus soal DELETE pq FROM PackageQuestions pq INNER JOIN AssessmentPackages ap ON pq.AssessmentPackageId = ap.Id WHERE ap.AssessmentSessionId IN (SELECT Id FROM @TestSessionIds); -- 9. Hapus paket DELETE FROM AssessmentPackages WHERE AssessmentSessionId IN (SELECT Id FROM @TestSessionIds); -- 10. Terakhir — hapus AssessmentSession DELETE FROM AssessmentSessions WHERE Id IN (SELECT Id FROM @TestSessionIds); -- 11. (Opsional) Bersihkan audit log terkait -- DELETE FROM AuditLogs WHERE Description LIKE '%[TEST]%'; -- Verify — harus return 0 row SELECT COUNT(*) AS SisaTestAssessment FROM AssessmentSessions WHERE Title LIKE '[TEST]%'; -- Jika verify OK: COMMIT TRANSACTION; -- Kalau ada masalah: ROLLBACK TRANSACTION;
